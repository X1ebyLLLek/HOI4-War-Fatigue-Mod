# =============================================
# plus_cs_crisis_step
# Main entry point for crisis calculations
# =============================================
plus_cs_crisis_step = {

	# 1. Civil War Risk Logic
	# FIX: Synergy crises (homefront_collapse, total_exhaustion) remove internal_crisis_high,
	# so civil wars never fired. Now we check synergy modifiers too.
	if = {
		limit = {
			OR = {
				has_idea = internal_crisis_high
				has_dynamic_modifier = homefront_collapse
				has_dynamic_modifier = total_exhaustion
			}
			NOT = { has_country_flag = civil_war_risk_active }

			OR = {
				AND = { has_government = democratic democratic < 0.5 }
				AND = { has_government = fascism fascism < 0.5 }
				AND = { has_government = communism communism < 0.5 }
				AND = { has_government = neutrality neutrality < 0.5 }
			}
			# CIVIL WAR GUARD
			OR = {
				has_civil_war = no
				has_country_flag = plus_cs_had_civil_war
			}
		}
		random_list = {
			98 = {}
			2 = { 
				# Launch Civil War Logic
				if = {
					limit = { has_civil_war = no }
					random_list = {
					50 = {
						# Размер мятежа зависит от оппозиции (мин 15%, макс 50%)
						set_temp_variable = { cw_size = 0.15 }
						if = { limit = { has_government = democratic fascism > 0.3 } set_temp_variable = { cw_size = 0.35 } }
						if = { limit = { has_government = democratic fascism > 0.4 } set_temp_variable = { cw_size = 0.45 } }
						if = { limit = { has_government = neutrality fascism > 0.3 } set_temp_variable = { cw_size = 0.35 } }
						if = { limit = { has_government = fascism democratic > 0.3 } set_temp_variable = { cw_size = 0.35 } }
						if = { limit = { has_government = communism democratic > 0.3 } set_temp_variable = { cw_size = 0.35 } }
						
						if = { limit = { has_government = democratic } start_civil_war = { ideology = fascism size = var:cw_size } }
						if = { limit = { has_government = neutrality } start_civil_war = { ideology = fascism size = var:cw_size } }
						if = { limit = { has_government = fascism } start_civil_war = { ideology = democratic size = var:cw_size } }
						if = { limit = { has_government = communism } start_civil_war = { ideology = democratic size = var:cw_size } }
					}
					50 = {
						set_temp_variable = { cw_size = 0.15 }
						if = { limit = { has_government = democratic communism > 0.3 } set_temp_variable = { cw_size = 0.35 } }
						if = { limit = { has_government = democratic communism > 0.4 } set_temp_variable = { cw_size = 0.45 } }
						if = { limit = { has_government = neutrality communism > 0.3 } set_temp_variable = { cw_size = 0.35 } }
						if = { limit = { has_government = fascism communism > 0.3 } set_temp_variable = { cw_size = 0.35 } }
						if = { limit = { has_government = communism fascism > 0.3 } set_temp_variable = { cw_size = 0.35 } }
						
						if = { limit = { has_government = democratic } start_civil_war = { ideology = communism size = var:cw_size } }
						if = { limit = { has_government = neutrality } start_civil_war = { ideology = communism size = var:cw_size } }
						if = { limit = { has_government = fascism } start_civil_war = { ideology = communism size = var:cw_size } }
						if = { limit = { has_government = communism } start_civil_war = { ideology = fascism size = var:cw_size } }
					}
				}
				}
				set_country_flag = { flag = civil_war_risk_active days = 30 }
				# Mark that this country had a civil war (enables post-civil war collapse events 11-13)
				set_country_flag = plus_cs_had_civil_war
			}
		}
	}
	
	# ==============================================================================
	# 2. REALISTIC WEEKLY IDEOLOGY DRIFT
	# Replaced monthly heavy drift with smoother weekly ticks.
	# Drift depends on current ideology and crisis level.
	# ==============================================================================
	if = {
		limit = { NOT = { AND = { date < 1938.1.1 has_civil_war = yes } } }
		
		# Define Base Drift Value based on Crisis Level
		set_temp_variable = { drift_tick = 0 }
		
		if = {
			limit = { has_idea = internal_crisis_high }
			set_temp_variable = { drift_tick = 0.002 } # 0.2% per week (~0.8% month)
		}
		else_if = {
			limit = { has_idea = internal_crisis_medium }
			set_temp_variable = { drift_tick = 0.001 } # 0.1% per week (~0.4% month)
		}
		else_if = {
			limit = { has_idea = internal_crisis_low }
			set_temp_variable = { drift_tick = 0.0005 } # 0.05% per week (~0.2% month)
		}

		# Execute Drift (Only if crisis active)
		if = {
			limit = { check_variable = { drift_tick > 0 } }
			
			# A. DEMOCRACIES -> Drift to Extremism ("Strong Hand" or "Revolution")
			if = {
				limit = { has_government = democratic }
				random_list = {
					# 50% Fascism (Order/Nationalism)
					50 = { add_popularity = { ideology = fascism popularity = var:drift_tick } }
					# 50% Communism (Equality/Protest)
					50 = { add_popularity = { ideology = communism popularity = var:drift_tick } }
				}
			}
			
			# B. COMMUNISM -> Drift to Democracy (Freedom) or Fascism (Reaction)
			else_if = {
				limit = { has_government = communism }
				random_list = {
					# 40% Democracy (Peace/Freedom)
					40 = { add_popularity = { ideology = democratic popularity = var:drift_tick } }
					# 40% Fascism (Civil War/Coup)
					40 = { add_popularity = { ideology = fascism popularity = var:drift_tick } }
					# 20% Neturality (Apathy)
					20 = { add_popularity = { ideology = neutrality popularity = var:drift_tick } }
				}
			}

			# C. FASCISM -> Drift to Democracy (Peace) or Communism (Resistance)
			else_if = {
				limit = { has_government = fascism }
				random_list = {
					# 40% Democracy (Peace/Freedom)
					40 = { add_popularity = { ideology = democratic popularity = var:drift_tick } }
					# 40% Communism (Partisans)
					40 = { add_popularity = { ideology = communism popularity = var:drift_tick } }
					# 20% Neturality (Apathy)
					20 = { add_popularity = { ideology = neutrality popularity = var:drift_tick } }
				}
			}
			
			# D. NEUTRALITY -> Radicalization in ALL directions
			else = {
				random_list = {
					# 33% Democracy (Reformers)
					33 = { add_popularity = { ideology = democratic popularity = var:drift_tick } }
					# 33% Fascism (Nationalists)
					33 = { add_popularity = { ideology = fascism popularity = var:drift_tick } }
					# 33% Communism (Socialists)
					33 = { add_popularity = { ideology = communism popularity = var:drift_tick } }
				}
			}
		}
	}

	# 3. Late War Fatigue (Vietnam Effect)
	if = {
		limit = {
			has_war = yes
			date > 1942.1.1
			NOT = { has_country_flag = plus_cs_exhausted }
		}
		
		set_temp_variable = { fatigue_mult = 1 }
		if = { limit = { date > 1944.1.1 } set_temp_variable = { fatigue_mult = 2 } }
		# NEW: Ultra-late war (post-1946) — extreme exhaustion
		if = { limit = { date > 1946.1.1 } set_temp_variable = { fatigue_mult = 3 } }
		
		# Base Stab Hit
		set_temp_variable = { stab_hit = -0.002 }
		multiply_temp_variable = { stab_hit = var:fatigue_mult }
		add_stability = var:stab_hit
		
		# Casualty Fatigue
		if = {
			limit = { casualties > 500000 }
			add_stability = var:stab_hit
		}
		
		# Democratic Sensitivity — war fatigue destabilizes democracies more
		if = {
			limit = { has_government = democratic }
			add_stability = var:stab_hit
			# Долгая война радикализирует население: буст крупнейшей оппозиции
			set_temp_variable = { war_pop_drift = 0.001 }
			multiply_temp_variable = { war_pop_drift = var:fatigue_mult }
			# Дрифт в сторону самой большой оппозиции
			if = {
				limit = { fascism > communism }
				add_popularity = { ideology = fascism popularity = var:war_pop_drift }
			}
			else = {
				add_popularity = { ideology = communism popularity = var:war_pop_drift }
			}
		}
	}
}

# =============================================
# plus_cs_internal_crisis_check_step
# Handles Ideology Crisis level management
# =============================================
plus_cs_internal_crisis_check_step = {
	# Определяем уровень кризиса
	set_temp_variable = { crisis_level = 0 }
	
	if = {
		limit = { 
			OR = { has_war_support < 0.2 has_stability < 0.2 }
			# CIVIL WAR GUARD
			OR = {
				has_civil_war = no
				has_country_flag = plus_cs_had_civil_war
			}
		}
		set_temp_variable = { crisis_level = 3 }
		add_timed_idea = { idea = internal_crisis_high days = 60 }
		remove_ideas = internal_crisis_medium
		remove_ideas = internal_crisis_low
	}
	else_if = {
		limit = { has_war = yes OR = { has_war_support < 0.4 has_stability < 0.4 } }
		set_temp_variable = { crisis_level = 2 }
		add_timed_idea = { idea = internal_crisis_medium days = 60 }
		remove_ideas = internal_crisis_high
		remove_ideas = internal_crisis_low
	}
	else_if = {
		limit = { has_war = yes OR = { has_war_support < 0.6 has_stability < 0.6 } }
		set_temp_variable = { crisis_level = 1 }
		add_timed_idea = { idea = internal_crisis_low days = 60 }
		remove_ideas = internal_crisis_high
		remove_ideas = internal_crisis_medium
	}
	else = {
		remove_ideas = internal_crisis_high
		remove_ideas = internal_crisis_medium
		remove_ideas = internal_crisis_low
	}
	
	# ========================================
	# [REMOVED] MONTHLY IDEOLOGY DRIFT
	# Logic moved to weekly 'plus_cs_crisis_step' for better granularity.
	# ========================================
}

# =============================================
# plus_cs_political_collapse_check
# Triggers collapse events when stability/WS < 10%
# =============================================
plus_cs_political_collapse_check = {
	# 1. PUPPET INSTANT CAPITULATION (5% Monthly Chance if Had Civil War)
	if = {
		limit = {
			has_war = yes
			is_puppet = yes
			has_country_flag = plus_cs_had_civil_war
		}
		# 2% Chance
		random_list = {
			2 = { country_event = { id = plus_cs_internal.12 } }
			98 = { }
		}
	}

	# 2. FREE NATION CRISIS DISPATCHER (Monthly Pulse)
	# This event (internal.10) selects between Coup, Paralysis, etc.
	# FIX: Coups can now also fire without prior civil war if crisis is extreme.
	# Historical precedent: July 20 plot, military juntas in collapsing regimes.
	if = {
		limit = {
			has_war = yes
			is_puppet = no
			NOT = { has_country_flag = plus_cs_collapse_cooldown }
			NOT = { has_civil_war = yes }
			# Path A: Post-civil-war collapse (original logic)
			# Path B: Extreme crisis without prior civil war (stab < 15%)
			OR = {
				has_country_flag = plus_cs_had_civil_war
				AND = {
					has_stability < 0.15
					OR = {
						has_dynamic_modifier = homefront_collapse
						has_dynamic_modifier = total_exhaustion
					}
				}
			}
		}
		
		# 2% chance monthly for these events
		random_list = {
			2 = {
				country_event = { id = plus_cs_internal.10 }
				# 30 day cooldown before next check
				set_country_flag = { flag = plus_cs_collapse_cooldown days = 30 }
			}
			98 = { }
		}
	}
	
	# Clean up Civil War Flag if peace returns
	if = {
		limit = { has_war = no has_country_flag = plus_cs_had_civil_war }
		clr_country_flag = plus_cs_had_civil_war
	}
	
	# Remove paralysis if conditions improve
	if = {
		limit = {
			has_idea = political_paralysis
			has_stability > 0.15
			has_war_support > 0.15
		}
		remove_ideas = political_paralysis
	}
}
