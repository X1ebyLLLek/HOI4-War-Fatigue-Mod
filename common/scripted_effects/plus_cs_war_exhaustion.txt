# =============================================
# plus_cs_heavy_calc_step
# Executes heavy calculations every ~28 days (tick-spread)
# =============================================
plus_cs_heavy_calc_step = {

	# --- INFLATION LOGIC ---
	if = {
		limit = {
			has_war = yes
			# CIVIL WAR GUARD
			OR = {
				has_civil_war = no
				has_country_flag = plus_cs_had_civil_war
			}
			OR = {
				has_stability < 0.20
				AND = {
					has_stability < 0.45
					OR = {
						surrender_progress > 0.10
						has_dynamic_modifier = manpower_strain_mild
						has_dynamic_modifier = manpower_strain_high
						has_war_support < 0.30
					}
				}
				has_dynamic_modifier = manpower_strain_high
				has_dynamic_modifier = manpower_strain_collapse
			}
		}
		add_dynamic_modifier = { modifier = war_inflation_crisis days = 35 }
		# R4: Динамический удар от инфляции — масштабируется от тяжести кризисов
		# P1 FIX: Halved base, reduced multipliers to prevent cascade
		set_temp_variable = { inflation_hit = -0.0015 }
		if = { limit = { has_dynamic_modifier = manpower_strain_collapse } multiply_temp_variable = { inflation_hit = 1.5 } }
		else_if = { limit = { has_dynamic_modifier = manpower_strain_high } multiply_temp_variable = { inflation_hit = 1.25 } }
		if = { limit = { has_stability < 0.2 } multiply_temp_variable = { inflation_hit = 1.25 } }
		add_stability = var:inflation_hit
		add_war_support = var:inflation_hit
	}
	else = {
		if = {
			limit = { has_dynamic_modifier = war_inflation_crisis }
			remove_dynamic_modifier = { modifier = war_inflation_crisis }
		}
	}

	# --- OFFICER CHAOS LOGIC ---
	if = {
		limit = { 
			OR = {
				NOT = { check_variable = { plus_cs_prev_battalions > 0 } }
				NOT = { num_divisions > 24 }
				check_variable = { plus_cs_prev_battalions < 10 }
			}
		}
		set_variable = { plus_cs_prev_battalions = num_divisions }
		set_variable = { plus_cs_prev_casualties = casualties }
	}
	else = {
		set_temp_variable = { officer_cap_calc = num_research_slots }
		multiply_temp_variable = { officer_cap_calc = 200 }
		add_to_temp_variable = { officer_cap_calc = 400 }
		
		if = { limit = { is_major = yes } add_to_temp_variable = { officer_cap_calc = 500 } }

		set_temp_variable = { total_strain = num_divisions }
		subtract_from_temp_variable = { total_strain = var:plus_cs_prev_battalions }
		clamp_temp_variable = { var = total_strain min = 0 }
		multiply_temp_variable = { total_strain = 0.5 } 

		set_temp_variable = { casualty_load = casualties }
		subtract_from_temp_variable = { casualty_load = var:plus_cs_prev_casualties }
		clamp_temp_variable = { var = casualty_load min = 0 }
		divide_temp_variable = { var = casualty_load value = 3500 }

		add_to_temp_variable = { var = total_strain value = var:casualty_load }

		if = { 
			limit = { num_divisions > 0 } 
			set_temp_variable = { div_count = num_divisions }
			divide_temp_variable = { var = total_strain value = div_count } 
		}

		if = {
			limit = { check_variable = { var = num_divisions value = var:officer_cap_calc compare = greater_than } }
			multiply_temp_variable = { var = total_strain value = 1.2 }
			set_temp_variable = { double_cap = var:officer_cap_calc }
			multiply_temp_variable = { var = double_cap value = 2 }
			if = { limit = { check_variable = { var = num_divisions value = var:double_cap compare = greater_than } } multiply_temp_variable = { var = total_strain value = 1.66 } }
		}

		if = {
			limit = { check_variable = { total_strain > 0.40 } }
			add_dynamic_modifier = { modifier = officer_shortage_collapse days = 60 }
			if = { limit = { has_dynamic_modifier = officer_shortage_high } remove_dynamic_modifier = { modifier = officer_shortage_high } }
			if = { limit = { has_dynamic_modifier = officer_shortage_mild } remove_dynamic_modifier = { modifier = officer_shortage_mild } }
		}
		else_if = {
			limit = { check_variable = { total_strain > 0.20 } }
			add_dynamic_modifier = { modifier = officer_shortage_high days = 60 }
			if = { limit = { has_dynamic_modifier = officer_shortage_collapse } remove_dynamic_modifier = { modifier = officer_shortage_collapse } }
			if = { limit = { has_dynamic_modifier = officer_shortage_mild } remove_dynamic_modifier = { modifier = officer_shortage_mild } }
		}
		else_if = {
			limit = { check_variable = { total_strain > 0.08 } }
			add_dynamic_modifier = { modifier = officer_shortage_mild days = 60 }
			if = { limit = { has_dynamic_modifier = officer_shortage_collapse } remove_dynamic_modifier = { modifier = officer_shortage_collapse } }
			if = { limit = { has_dynamic_modifier = officer_shortage_high } remove_dynamic_modifier = { modifier = officer_shortage_high } }
		}
		else = {
			if = { limit = { has_dynamic_modifier = officer_shortage_collapse } remove_dynamic_modifier = { modifier = officer_shortage_collapse } }
			if = { limit = { has_dynamic_modifier = officer_shortage_high } remove_dynamic_modifier = { modifier = officer_shortage_high } }
			if = { limit = { has_dynamic_modifier = officer_shortage_mild } remove_dynamic_modifier = { modifier = officer_shortage_mild } }
		}
		set_variable = { plus_cs_prev_battalions = num_divisions }
		set_variable = { plus_cs_prev_casualties = casualties }
	}
	
	# --- MANPOWER CRISIS LOGIC ---
	if = {
		limit = { has_war = yes }
		add_to_variable = { plus_cs_pop_check_timer = 1 }
		if = {
			limit = {
				OR = {
					# Cache Population every ~560 days (80 ticks × 7 days)
					check_variable = { plus_cs_pop_check_timer > 80 } 
					check_variable = { plus_cs_need_recalc = 1 }
					check_variable = { core_pop_calc = 0 }
				}
			}
			set_variable = { plus_cs_pop_check_timer = 0 }
			set_variable = { plus_cs_need_recalc = 0 }
			
			# OPTIMIZATION: Use country variable instead of temp for caching
			set_variable = { core_pop_calc = 0 }
			
			every_owned_state = {
				if = { limit = { is_core_of = ROOT } add_to_variable = { var = ROOT.core_pop_calc value = state_population } }
				else = {
					set_temp_variable = { local_pop = state_population }
					multiply_temp_variable = { var = local_pop value = 0.15 }
					add_to_variable = { var = ROOT.core_pop_calc value = local_pop }
				}
			}
			
			# OPT: Flat approximation for subject population
			# Old code: every_subject_country { every_owned_state } = O(N×M)
			# New code: count subjects × fraction of own pop = O(N)
			set_temp_variable = { subj_count = 0 }
			every_subject_country = {
				add_to_temp_variable = { subj_count = 1 }
			}
			if = {
				limit = { check_variable = { subj_count > 0 } }
				set_temp_variable = { subj_bonus = core_pop_calc }
				multiply_temp_variable = { subj_bonus = 0.15 }
				multiply_temp_variable = { subj_bonus = subj_count }
				add_to_variable = { core_pop_calc = subj_bonus }
			}
		}

		set_temp_variable = { total_drain = num_divisions }
		multiply_temp_variable = { var = total_drain value = 1000 }
		add_to_temp_variable = { var = total_drain value = casualties }
		set_temp_variable = { limit_base = var:core_pop_calc }
		
		# R6: Идеологический множитель порогов (else_if, mutually exclusive)
		# Демократии чуть быстрее впадают в кризис, тоталитарные — терпят дольше
		set_temp_variable = { manpower_ideo_mult = 1.0 }
		if = { limit = { has_government = democratic } set_temp_variable = { manpower_ideo_mult = 0.85 } }
		else_if = { limit = { OR = { has_government = fascism has_government = communism } } set_temp_variable = { manpower_ideo_mult = 1.3 } }
		

		# Collapse (> 5% × ideology mult of core pop)
		set_temp_variable = { limit_check = var:limit_base }
		set_temp_variable = { crisis_threshold = 0.05 }
		multiply_temp_variable = { crisis_threshold = manpower_ideo_mult }
		multiply_temp_variable = { var = limit_check value = var:crisis_threshold }

		if = {
			limit = { check_variable = { var = total_drain value = var:limit_check compare = greater_than } }
			add_dynamic_modifier = { modifier = manpower_strain_collapse days = 35 }
			if = { limit = { has_dynamic_modifier = manpower_strain_high } remove_dynamic_modifier = { modifier = manpower_strain_high } }
			if = { limit = { has_dynamic_modifier = manpower_strain_mild } remove_dynamic_modifier = { modifier = manpower_strain_mild } }
			# R1: set_country_flag не поддерживает var: — используем if/else_if
			if = { limit = { check_variable = { plus_cs_pressure_ratio > 1.5 } } set_country_flag = { flag = plus_cs_exhausted days = 60 } }
			else_if = { limit = { check_variable = { plus_cs_pressure_ratio > 1.0 } } set_country_flag = { flag = plus_cs_exhausted days = 45 } }
			else = { set_country_flag = { flag = plus_cs_exhausted days = 28 } }
			clr_country_flag = plus_cs_war_strain
		}
		else = {
			# High (> 3% × ideology mult of core pop)
			set_temp_variable = { limit_check = var:limit_base }
			set_temp_variable = { crisis_threshold = 0.03 }
			multiply_temp_variable = { crisis_threshold = manpower_ideo_mult }
			multiply_temp_variable = { var = limit_check value = var:crisis_threshold }
			if = {
				limit = { check_variable = { var = total_drain value = var:limit_check compare = greater_than } }
				add_dynamic_modifier = { modifier = manpower_strain_high days = 35 }
				if = { limit = { has_dynamic_modifier = manpower_strain_collapse } remove_dynamic_modifier = { modifier = manpower_strain_collapse } }
				if = { limit = { has_dynamic_modifier = manpower_strain_mild } remove_dynamic_modifier = { modifier = manpower_strain_mild } }
				if = { limit = { check_variable = { plus_cs_pressure_ratio > 1.5 } } set_country_flag = { flag = plus_cs_exhausted days = 60 } }
				else_if = { limit = { check_variable = { plus_cs_pressure_ratio > 1.0 } } set_country_flag = { flag = plus_cs_exhausted days = 45 } }
				else = { set_country_flag = { flag = plus_cs_exhausted days = 28 } }
				clr_country_flag = plus_cs_war_strain
			}
			else = {
				# Mild (> 1.5% × ideology mult of core pop)
				set_temp_variable = { limit_check = var:limit_base }
				set_temp_variable = { crisis_threshold = 0.015 }
				multiply_temp_variable = { crisis_threshold = manpower_ideo_mult }
				multiply_temp_variable = { var = limit_check value = var:crisis_threshold }
				if = {
					limit = { check_variable = { var = total_drain value = var:limit_check compare = greater_than } }
					add_dynamic_modifier = { modifier = manpower_strain_mild days = 35 }
					if = { limit = { has_dynamic_modifier = manpower_strain_collapse } remove_dynamic_modifier = { modifier = manpower_strain_collapse } }
					if = { limit = { has_dynamic_modifier = manpower_strain_high } remove_dynamic_modifier = { modifier = manpower_strain_high } }
					if = { limit = { check_variable = { plus_cs_pressure_ratio > 1.5 } } set_country_flag = { flag = plus_cs_war_strain days = 60 } }
					else_if = { limit = { check_variable = { plus_cs_pressure_ratio > 1.0 } } set_country_flag = { flag = plus_cs_war_strain days = 45 } }
					else = { set_country_flag = { flag = plus_cs_war_strain days = 28 } }
				}
				else = {
					if = { limit = { has_dynamic_modifier = manpower_strain_collapse } remove_dynamic_modifier = { modifier = manpower_strain_collapse } }
					if = { limit = { has_dynamic_modifier = manpower_strain_high } remove_dynamic_modifier = { modifier = manpower_strain_high } }
					if = { limit = { has_dynamic_modifier = manpower_strain_mild } remove_dynamic_modifier = { modifier = manpower_strain_mild } }
				}
			}
		}
		if = {
			limit = { 
				is_major = no
				has_war_with_major = yes
				surrender_progress > 0.05
				any_owned_state = { is_core_of = ROOT is_controlled_by = ROOT }
			}
			add_timed_idea = { idea = desperate_defense_effort days = 35 }
		}
		else = { remove_ideas = desperate_defense_effort }
	}
	else = {
		if = { limit = { has_dynamic_modifier = manpower_strain_mild } remove_dynamic_modifier = { modifier = manpower_strain_mild } }
		if = { limit = { has_dynamic_modifier = manpower_strain_high } remove_dynamic_modifier = { modifier = manpower_strain_high } }
		if = { limit = { has_dynamic_modifier = manpower_strain_collapse } remove_dynamic_modifier = { modifier = manpower_strain_collapse } }
		remove_ideas = desperate_defense_effort
		# Также убираем комбинированные кризисы при мире
		if = { limit = { has_dynamic_modifier = army_cohesion_crisis } remove_dynamic_modifier = { modifier = army_cohesion_crisis } }
		if = { limit = { has_dynamic_modifier = homefront_collapse } remove_dynamic_modifier = { modifier = homefront_collapse } }
		if = { limit = { has_dynamic_modifier = total_exhaustion } remove_dynamic_modifier = { modifier = total_exhaustion } }
	}
}

# =============================================
# plus_cs_synergy_crisis_check
# Проверяет комбинации кризисов и применяет
# синергетические модификаторы
# ВАЖНО: Комбинированные кризисы ЗАМЕНЯЮТ базовые,
# а не суммируются с ними!
# =============================================
plus_cs_synergy_crisis_check = {
	# Считаем количество серьёзных кризисов ДО замены
	set_temp_variable = { crisis_count = 0 }
	
	# Флаги для отслеживания активных кризисов
	set_temp_variable = { has_manpower_crisis = 0 }
	set_temp_variable = { has_officer_crisis = 0 }
	set_temp_variable = { has_internal_crisis = 0 }
	set_temp_variable = { has_inflation_crisis = 0 }
	
	# Manpower Strain High/Collapse
	if = {
		limit = {
			OR = {
				has_dynamic_modifier = manpower_strain_high
				has_dynamic_modifier = manpower_strain_collapse
			}
		}
		set_temp_variable = { has_manpower_crisis = 1 }
		add_to_temp_variable = { crisis_count = 1 }
	}
	
	# Officer Shortage High/Collapse
	if = {
		limit = {
			OR = {
				has_dynamic_modifier = officer_shortage_high
				has_dynamic_modifier = officer_shortage_collapse
			}
		}
		set_temp_variable = { has_officer_crisis = 1 }
		add_to_temp_variable = { crisis_count = 1 }
	}
	
	# Internal Crisis High
	if = {
		limit = { has_idea = internal_crisis_high }
		set_temp_variable = { has_internal_crisis = 1 }
		add_to_temp_variable = { crisis_count = 1 }
	}
	
	# War Inflation
	if = {
		limit = { has_dynamic_modifier = war_inflation_crisis }
		set_temp_variable = { has_inflation_crisis = 1 }
		add_to_temp_variable = { crisis_count = 1 }
	}
	
	# ========================================
	# TOTAL EXHAUSTION (проверяем первым!)
	# Триггер: 3+ серьёзных кризиса одновременно
	# ЗАМЕНЯЕТ ВСЕ базовые кризисы
	# ========================================
	if = {
		limit = {
			check_variable = { crisis_count > 2 }
		}
		# Применяем Total Exhaustion
		add_dynamic_modifier = { modifier = total_exhaustion days = 35 }
		set_country_flag = { flag = plus_cs_total_exhaustion days = 35 }
		
		# УБИРАЕМ ВСЕ базовые кризисы (они поглощены) — OPT: с проверками
		if = { limit = { has_dynamic_modifier = manpower_strain_mild } remove_dynamic_modifier = { modifier = manpower_strain_mild } }
		if = { limit = { has_dynamic_modifier = manpower_strain_high } remove_dynamic_modifier = { modifier = manpower_strain_high } }
		if = { limit = { has_dynamic_modifier = manpower_strain_collapse } remove_dynamic_modifier = { modifier = manpower_strain_collapse } }
		if = { limit = { has_dynamic_modifier = officer_shortage_mild } remove_dynamic_modifier = { modifier = officer_shortage_mild } }
		if = { limit = { has_dynamic_modifier = officer_shortage_high } remove_dynamic_modifier = { modifier = officer_shortage_high } }
		if = { limit = { has_dynamic_modifier = officer_shortage_collapse } remove_dynamic_modifier = { modifier = officer_shortage_collapse } }
		if = { limit = { has_dynamic_modifier = war_inflation_crisis } remove_dynamic_modifier = { modifier = war_inflation_crisis } }
		remove_ideas = internal_crisis_high
		# Убираем также подкризисы
		if = { limit = { has_dynamic_modifier = army_cohesion_crisis } remove_dynamic_modifier = { modifier = army_cohesion_crisis } }
		if = { limit = { has_dynamic_modifier = homefront_collapse } remove_dynamic_modifier = { modifier = homefront_collapse } }
		
		# TOTAL EXHAUSTION = МАКСИМАЛЬНЫЙ ДРИФТ К ОППОЗИЦИИ (5% за тик)
		set_temp_variable = { drift_value = 0.05 }
		if = {
			limit = { has_government = fascism }
			add_popularity = { ideology = democratic popularity = drift_value }
			add_popularity = { ideology = communism popularity = drift_value }
		}
		else_if = {
			limit = { has_government = communism }
			add_popularity = { ideology = democratic popularity = drift_value }
		}
		else_if = {
			limit = { has_government = democratic }
			set_temp_variable = { drift_value = 0.025 }
			add_popularity = { ideology = fascism popularity = drift_value }
			add_popularity = { ideology = communism popularity = drift_value }
		}
		else = {
			add_popularity = { ideology = democratic popularity = drift_value }
		}
	}
	else = {
		if = { limit = { has_dynamic_modifier = total_exhaustion } remove_dynamic_modifier = { modifier = total_exhaustion } }
		clr_country_flag = plus_cs_total_exhaustion
		
		# ========================================
		# ARMY COHESION CRISIS
		# Триггер: Manpower High/Collapse + Officer High/Collapse
		# ЗАМЕНЯЕТ: officer_shortage_*, manpower_strain_*
		# ========================================
		if = {
			limit = {
				check_variable = { has_manpower_crisis = 1 }
				check_variable = { has_officer_crisis = 1 }
			}
			# Применяем Army Cohesion Crisis
			add_dynamic_modifier = { modifier = army_cohesion_crisis days = 35 }
			
			# УБИРАЕМ базовые кризисы армии (они поглощены) — OPT: с проверками
			if = { limit = { has_dynamic_modifier = manpower_strain_mild } remove_dynamic_modifier = { modifier = manpower_strain_mild } }
			if = { limit = { has_dynamic_modifier = manpower_strain_high } remove_dynamic_modifier = { modifier = manpower_strain_high } }
			if = { limit = { has_dynamic_modifier = manpower_strain_collapse } remove_dynamic_modifier = { modifier = manpower_strain_collapse } }
			if = { limit = { has_dynamic_modifier = officer_shortage_mild } remove_dynamic_modifier = { modifier = officer_shortage_mild } }
			if = { limit = { has_dynamic_modifier = officer_shortage_high } remove_dynamic_modifier = { modifier = officer_shortage_high } }
			if = { limit = { has_dynamic_modifier = officer_shortage_collapse } remove_dynamic_modifier = { modifier = officer_shortage_collapse } }
		}
		else = {
			if = { limit = { has_dynamic_modifier = army_cohesion_crisis } remove_dynamic_modifier = { modifier = army_cohesion_crisis } }
		}
		
		# ========================================
		# HOMEFRONT COLLAPSE
		# Триггер: War Inflation + Internal Crisis High
		# ЗАМЕНЯЕТ: war_inflation_crisis, internal_crisis_high
		# ========================================
		if = {
			limit = {
				check_variable = { has_inflation_crisis = 1 }
				check_variable = { has_internal_crisis = 1 }
				# FIX: Only trigger homefront collapse when actually losing territory
				surrender_progress > 0.15
			}
			# Применяем Homefront Collapse
			add_dynamic_modifier = { modifier = homefront_collapse days = 35 }
			
			# УБИРАЕМ базовые кризисы тыла (они поглощены)
			if = { limit = { has_dynamic_modifier = war_inflation_crisis } remove_dynamic_modifier = { modifier = war_inflation_crisis } }
			remove_ideas = internal_crisis_high
			
			# HOMEFRONT COLLAPSE = СИЛЬНЫЙ ДРИФТ К ОППОЗИЦИИ (3% за тик)
			set_temp_variable = { drift_value = 0.03 }
			if = {
				limit = { has_government = fascism }
				add_popularity = { ideology = democratic popularity = drift_value }
				add_popularity = { ideology = communism popularity = drift_value }
			}
			else_if = {
				limit = { has_government = communism }
				add_popularity = { ideology = democratic popularity = drift_value }
			}
			else_if = {
				limit = { has_government = democratic }
				set_temp_variable = { drift_value = 0.015 }
				add_popularity = { ideology = fascism popularity = drift_value }
				add_popularity = { ideology = communism popularity = drift_value }
			}
			else = {
				add_popularity = { ideology = democratic popularity = drift_value }
			}
		}
		else = {
			if = { limit = { has_dynamic_modifier = homefront_collapse } remove_dynamic_modifier = { modifier = homefront_collapse } }
		}
	}
}

# =============================================
# plus_cs_cleanup_crisis_modifiers
# Reusable cleanup for all crisis modifiers/ideas.
# Called after peace, coups, capitulation events.
# OPT: Extracted from 3 duplicate blocks in events.
# =============================================
plus_cs_cleanup_crisis_modifiers = {
	# OPT: Fast path — skip all removes if country was never in crisis
	if = {
		limit = {
			NOT = { has_country_flag = plus_cs_exhausted }
			NOT = { has_country_flag = plus_cs_war_strain }
			NOT = { has_idea = internal_crisis_high }
			NOT = { has_idea = internal_crisis_medium }
			NOT = { has_idea = internal_crisis_low }
		}
		# No crisis markers → nothing to clean up
	}
	else = {
	remove_ideas = internal_crisis_high
	remove_ideas = internal_crisis_medium
	remove_ideas = internal_crisis_low
	remove_ideas = political_paralysis
	remove_ideas = desperate_defense_effort
	remove_ideas = panic_retreat_shock
	if = { limit = { has_dynamic_modifier = homefront_collapse } remove_dynamic_modifier = { modifier = homefront_collapse } }
	if = { limit = { has_dynamic_modifier = total_exhaustion } remove_dynamic_modifier = { modifier = total_exhaustion } }
