scripted_diplomatic_actions = {
	plus_cp_demand_core_states = {
		requires_acceptance = no
		cost = 0 
			
		# root is the initiator of action and this is the target country
		allowed = { 
			always = yes
		}
			
		visible = { 
			# Always visible in the list to avoid confusion, but greyed out if requirements aren't met
			always = yes
		}
			
		selectable = { 
			custom_trigger_tooltip = {
				tooltip = plus_cs_demand_not_core_tt
				# Action can only be sent if they control at least one state that is a core or claim of ours
				OR = {
					any_controlled_state = { is_core_of = ROOT }
					any_controlled_state = { is_claimed_by = ROOT }
				}
			}

			custom_trigger_tooltip = {
				tooltip = plus_cs_demand_not_at_war_tt
				ROOT = {
					has_war = no
					NOT = { has_civil_war = yes }
				}
				NOT = { has_war_with = ROOT }
				NOT = { has_civil_war = yes }
			}
			
			custom_trigger_tooltip = {
				tooltip = plus_cs_demand_not_faction_tt
				NOT = { is_in_faction_with = ROOT }
				NOT = { is_subject_of = ROOT }
				ROOT = { NOT = { is_subject_of = THIS } }
			}
			
			custom_trigger_tooltip = {
				tooltip = plus_cs_demand_no_recent_tt
				NOT = { has_country_flag = plus_cs_recent_demand }
			}
			
			custom_trigger_tooltip = {
				tooltip = plus_cs_demand_full_annexation_tt
				# Prevent full annexation by checking if target has at least one state that is NOT sender's core
				any_controlled_state = { NOT = { is_core_of = ROOT } }
			}

			ROOT = {
				# Democracies require 100% WT unless at war
				OR = {
					NOT = { has_government = democratic }
					threat > 0.99
					has_war = yes
				}
				# Non-Aligned require 50% WT unless at war
				OR = {
					NOT = { has_government = neutrality }
					threat > 0.49
					has_war = yes
				}
				# Fascists and Communists require 25% WT unless at war
				OR = {
					AND = {
						NOT = { has_government = fascism }
						NOT = { has_government = communism }
					}
					threat > 0.24
					has_war = yes
				}
			}
		}
			
		can_be_sent = { 
			always = yes
		}

		complete_effect = {
			set_country_flag = { flag = plus_cs_recent_demand days = 180 }
			country_event = { id = plus_cs_demands.1 days = 2 }
		}

		ai_desire = {
			base = 0
			modifier = {
				add = 100
				# AI will only try this if they are significantly stronger or equal
				ROOT = { strength_ratio = { tag = THIS ratio > 0.9 } }
					
				ROOT = {
					OR = {
						has_government = fascism
						has_government = communism
						has_country_leader_with_trait = dictator
						has_country_leader_with_trait = plus_cs_spirit_fanatic
					}
				}
			}
			# AI will NOT try this if they are already losing a war
			modifier = {
				factor = 0
				ROOT = { has_war = yes }
			}
		}
	}
}
