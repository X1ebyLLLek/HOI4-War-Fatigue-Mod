

# 1. РЕШЕНИЯ ВОЕННОГО ВРЕМЕНИ (CRISIS MANAGEMENT)
decision_category_crisis_management = {
    
    # Снятие паники от потери городов (против духа panic_retreat_shock)
    decree_stop_panic = {
        icon = GFX_decision_generic_army_support
        allowed = { always = yes }
        cost = 100
        days_re_enable = 60
        
        available = {
            has_war = yes
            has_idea = panic_retreat_shock
        }
        
        visible = {
            has_idea = panic_retreat_shock
        }
        
        complete_effect = {
            add_political_power = -100
            remove_ideas = panic_retreat_shock
            add_stability = 0.05
        }
        
        ai_will_do = {
            base = 100
            
            modifier = {
                factor = 200
                has_stability < 0.20
            }
            modifier = {
                factor = 50
                has_stability < 0.40
                has_stability > 0.19
            }
            modifier = {
                factor = 5
                has_stability < 0.60
                has_stability > 0.39
            }

            modifier = {
                factor = 2
                surrender_progress > 0.1
            }
        }
    }

    # Пропаганда для снижения внутреннего кризиса (против internal_crisis_*)
    decree_internal_stability = {
        icon = GFX_decision_generic_propaganda
        allowed = { always = yes }
        cost = 150
        days_re_enable = 120
        
        available = {
            has_war = yes
            OR = {
                has_idea = internal_crisis_high
                has_idea = internal_crisis_medium
                has_idea = internal_crisis_low
            }
        }
        
        visible = {
            OR = {
                has_idea = internal_crisis_high
                has_idea = internal_crisis_medium
                has_idea = internal_crisis_low
            }
        }
        
        complete_effect = {
            add_political_power = -150
            
            if = {
                limit = { has_idea = internal_crisis_high }
                remove_ideas = internal_crisis_high
                add_timed_idea = { idea = internal_crisis_medium days = 90 }
            }
            else_if = {
                limit = { has_idea = internal_crisis_medium }
                remove_ideas = internal_crisis_medium
                add_timed_idea = { idea = internal_crisis_low days = 90 }
            }
            else_if = {
                limit = { has_idea = internal_crisis_low }
                remove_ideas = internal_crisis_low
            }
        }

        ai_will_do = {
            base = 100
            
            modifier = {
                factor = 200
                OR = { has_stability < 0.20 has_war_support < 0.20 }
            }
            modifier = {
                factor = 50
                OR = { 
                    AND = { has_stability < 0.40 has_stability > 0.19 }
                    AND = { has_war_support < 0.40 has_war_support > 0.19 }
                }
            }
            modifier = {
                factor = 5
                OR = { 
                    AND = { has_stability < 0.60 has_stability > 0.39 }
                    AND = { has_war_support < 0.60 has_war_support > 0.39 }
                }
            }
        }
    }

    # Стабилизация правительства — снятие политического паралича
    decree_stabilize_government = {
        icon = GFX_decision_generic_political_discourse
        allowed = { always = yes }
        cost = 50
        days_re_enable = 90
        
        available = {
            has_war = yes
            has_idea = political_paralysis
        }
        
        visible = {
            has_idea = political_paralysis
        }
        
        complete_effect = {
            add_political_power = -50
            remove_ideas = political_paralysis
            add_stability = 0.05
            add_war_support = 0.05
        }
        
        ai_will_do = {
            base = 200
        }
    }
}
