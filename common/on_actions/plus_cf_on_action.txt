on_actions = {
	on_startup = {
		effect = {
			random_country = {
				limit = { 
					is_ai = no 
					is_dynamic_country = no 
				}
				country_event = { id = plus_cs.3 }
			}
			
			# FIX: Restored global.countries — used by plus_cp_subject_decisions.txt
			# Removed only global.province_controllers (unused, only in comments)
			clear_array = global.countries
			
			every_country = {
				add_to_array = { global.countries = THIS }
			}
			
			every_country = {
				plus_cs_assign_fighting_spirit = yes # Assign initial personality
				plus_cs_init_hidden_resolve = yes    # Assign hidden random variance (-50..+50)
				set_variable = { plus_cs_pop_check_timer = 10 }
				set_variable = { plus_cs_need_recalc = 1 }
				set_variable = { core_pop_calc = 0 }
				
				# Инициализация переменных
				# FIX: num_battalions нельзя записать в переменную. Заменено на num_divisions.
				set_variable = { plus_cs_prev_battalions = num_divisions }
				set_variable = { plus_cs_prev_casualties = casualties }
				set_variable = { economy_overheat_counter = 0 }
				set_variable = { plus_cs_prev_econ_law = 0 }
				
				# Новый таймер для пата
				set_variable = { plus_cs_stalemate_timer = 0 }
				
				# Start Optimization Loop
				# Delayed from 1 to 14 days to prevent day-1 calculation lag and spam
				country_event = { id = plus_cs.999 days = 14 } 
				
				# Global Startup Cooldown: Blocks all peace/submission actions for 30 days
				if = {
					limit = { NOT = { has_global_flag = plus_cs_mod_startup_cooldown } }
					set_global_flag = { flag = plus_cs_mod_startup_cooldown days = 30 }
				}
			}
		}
	}

	on_war_relation_added = {
		effect = {
			# BUGFIX 1: If this country signed a separate peace and has a truce with
			# the new enemy, immediately white_peace — prevents overlords/faction leaders
			# from dragging them back into a war they already left.
			if = {
				limit = {
					has_country_flag = plus_cs_signed_separate_peace
					has_truce_with = FROM
				}
				white_peace = FROM
			}

			set_variable = { plus_cs_war_count = 0 }
			every_enemy_country = {
				add_to_variable = { plus_cs_war_count = 1 }
			}
			
			# Assign Fighting Spirit (Organic Variability)
			plus_cs_assign_fighting_spirit = yes
			every_enemy_country = {
				plus_cs_assign_fighting_spirit = yes
			}
			
			# Инициализация переменных для новых войн
			set_variable = { vp@FROM = 0 }
			set_variable = { FROM.vp@ROOT = 0 }
			set_variable = { plus_cs_last_month_vp = 0 }
			set_variable = { plus_cs_stalemate_timer = 0 }
			
			# War Duration Tracking (permanent flag, checked by plus_cs_calc_ai_situation)
			if = {
				limit = { NOT = { has_country_flag = plus_cs_at_war_start } }
				set_country_flag = plus_cs_at_war_start
			}
			
			set_variable = { plus_cs_need_recalc = 1 }
			set_variable = { economy_overheat_counter = 0 }
		}
	}

	on_capitulation = {
		effect = {
			if = {
				limit = { is_major = yes }
				add_threat = -5
			}
			else_if = { 
				limit = { has_offensive_war = yes }
				add_threat = -1
			}
		}
	}

	on_peace = {
		effect = {
			if = {
				limit = { is_major = yes }
				add_threat = -3
			}
			else = {
				add_threat = -0.5
			}

			if = {
				limit = {
					threat > 0.2
					NOT = { 
						any_country = {
							is_major = yes
							has_war = yes
						}
					} 
				}
				add_threat = -1
			}

			set_variable = { plus_cs_last_month_vp = 0 }
			set_variable = { plus_cs_stalemate_timer = 0 }

			if = {
				limit = { has_war = no }
				
				# Очистка флагов
				clr_country_flag = plus_cs_urgent_peace
				clr_country_flag = plus_cs_exhausted
				clr_country_flag = plus_cs_stalemate
				clr_country_flag = plus_cs_war_strain
				clr_country_flag = civil_war_risk_active
				clr_country_flag = plus_cs_revolter_processed
				clr_country_flag = plus_cs_at_war_start
				clr_country_flag = plus_cs_offered_peace_before
				clr_country_flag = plus_cs_offered_peace_twice
				clr_country_flag = plus_cs_collapse_cooldown

				# Удаление идей 
				remove_ideas = internal_crisis_low
				remove_ideas = internal_crisis_medium
				remove_ideas = internal_crisis_high
				remove_ideas = desperate_defense_effort
				remove_ideas = fight_for_survival
				remove_ideas = panic_retreat_shock
				remove_ideas = political_paralysis

				# Очистка модификаторов (с проверками для оптимизации)
				if = { limit = { has_dynamic_modifier = war_inflation_crisis } remove_dynamic_modifier = { modifier = war_inflation_crisis } }
				if = { limit = { has_dynamic_modifier = manpower_strain_mild } remove_dynamic_modifier = { modifier = manpower_strain_mild } }
				if = { limit = { has_dynamic_modifier = manpower_strain_high } remove_dynamic_modifier = { modifier = manpower_strain_high } }
				if = { limit = { has_dynamic_modifier = manpower_strain_collapse } remove_dynamic_modifier = { modifier = manpower_strain_collapse } }
				if = { limit = { has_dynamic_modifier = officer_shortage_mild } remove_dynamic_modifier = { modifier = officer_shortage_mild } }
				if = { limit = { has_dynamic_modifier = officer_shortage_high } remove_dynamic_modifier = { modifier = officer_shortage_high } }
				if = { limit = { has_dynamic_modifier = officer_shortage_collapse } remove_dynamic_modifier = { modifier = officer_shortage_collapse } }
				
				# FIX: Очистка синергетических кризисов (заменяют базовые, но не чистились)
				if = { limit = { has_dynamic_modifier = army_cohesion_crisis } remove_dynamic_modifier = { modifier = army_cohesion_crisis } }
				if = { limit = { has_dynamic_modifier = homefront_collapse } remove_dynamic_modifier = { modifier = homefront_collapse } }
				if = { limit = { has_dynamic_modifier = total_exhaustion } remove_dynamic_modifier = { modifier = total_exhaustion } }
				
				# Очистка шока экономики
				if = { limit = { has_dynamic_modifier = economic_retooling_shock_minor } remove_dynamic_modifier = { modifier = economic_retooling_shock_minor } }
				if = { limit = { has_dynamic_modifier = economic_retooling_shock_medium } remove_dynamic_modifier = { modifier = economic_retooling_shock_medium } }
				if = { limit = { has_dynamic_modifier = economic_retooling_shock_major } remove_dynamic_modifier = { modifier = economic_retooling_shock_major } }
			}
		}
	}

	on_monthly = {
		effect = {
			
			# --- Пассивное остывание мира ---
			if = {
				limit = {
					original_tag = GER # Якорь: один мажор проверяет глобальный threat раз в месяц
					threat > 0.05
				}
				
				set_temp_variable = { world_is_burning = 0 }
				
				if = {
					limit = {
						any_country = {
							is_major = yes
							has_war = yes
						}
					}
					set_temp_variable = { world_is_burning = 1 }
				}
				
				if = {
					limit = { check_variable = { world_is_burning = 0 } }
					add_threat = -0.002
				}
			}

			# ==========================================================
			# MOVED TO TICK SPREADING (plus_cs_optimization.txt)
			# ==========================================================
			# Logic for Inflation, Officer Chaos, Economy Inertia, and Manpower Crisis
			# has been moved to the distributed loop to prevent 1st-of-month lag.


			# ==========================================================
			# 5. POLITICS & STALEMATE (MOVED TO plus_cs_optimization.txt)
			# ==========================================================
			# This logic is now handled in the distributed loop to prevent 
			# day-1 lag spikes.
			
		}
	}

	# --- OPTIMIZATION HOOKS (Tick-Spreading) ---
	
	on_daily = {
		effect = {
			# Fail-safe: Если страна "выпала" из цикла или появилась только что
			# OPT: Якорь на GER — проверяем только от одной страны вместо всех ~80
			if = {
				limit = {
					OR = {
						is_ai = no
						original_tag = GER
					}
				}
				random_list = {
					3 = {
						random_country = {
							limit = {
								NOT = { has_country_flag = plus_cs_loop_active }
								exists = yes
							}
							country_event = { id = plus_cs.999 }
						}
					}
					97 = { }
				}
			}
		}
	}

	on_state_control_changed = {
		effect = {
			# EVENT-DRIVEN STALEMATE RESET:
			# If control changes, the front is moving. Reset stalemate timers.
			ROOT = { set_variable = { plus_cs_stalemate_timer = 0 } }
			FROM = { set_variable = { plus_cs_stalemate_timer = 0 } }
			
			if = {
				limit = {
					FROM.FROM = { state_strategic_value > 15 is_core_of = ROOT }
					ROOT = { has_war = yes }
				}
				ROOT = {
					remove_ideas = panic_retreat_shock
					add_timed_idea = { idea = panic_retreat_shock days = 10 }
				}
			}
			
			# OPT: Debounce scorched earth — max once per 7 days per state
			if = {
				limit = {
					FROM.FROM = { is_core_of = FROM }
					FROM = { has_idea = fight_for_survival }
					ROOT = { has_war_with = FROM }
					NOT = { FROM.FROM = { has_state_flag = plus_cs_scorched_cd } }
				}
				FROM.FROM = {
					set_state_flag = { flag = plus_cs_scorched_cd days = 7 }
					damage_building = { type = infrastructure damage = 1.0 }
					damage_building = { type = supply_node damage = 1.0 }
					damage_building = { type = rail_way damage = 1.0 }
					random_list = {
						50 = { damage_building = { type = arms_factory damage = 0.5 } damage_building = { type = industrial_complex damage = 0.5 } }
						50 = { }
					}
				}
			}

			if = { limit = { FROM.FROM = { is_core_of = ROOT } } set_variable = { plus_cs_need_recalc = 1 } }
			if = { limit = { FROM.FROM = { is_core_of = FROM } } FROM = { set_variable = { plus_cs_need_recalc = 1 } } }

			# --- ОПТИМИЗАЦИЯ: НЕ СЧИТАЕМ, ЕСЛИ ПРОВИНЦИЯ ПУСТАЯ ---
			if = {
				limit = { FROM.FROM = { state_strategic_value > 0 } }
				
				set_temp_variable = { state_vp = FROM.FROM.state_strategic_value }
				set_temp_variable = { tot_pts = 0 }
				
				# OPTIMIZATION v4: THROTTLED VP CALCULATION
				# Instead of calculating VP on EVERY state control change (which can be 
				# hundreds per day in large battles), we use a simplified "winner takes all" 
				# approach for immediate feedback, and defer accurate calculation to the 
				# optimization loop.
				
				# SIMPLIFIED APPROACH: Direct VP transfer based on who captured
				# ROOT = new controller (winner), FROM = old controller (loser)
				# Winner gets VP points, loser loses them
				
				if = {
					limit = { ROOT = { has_war_with = FROM } }
					# Direct VP transfer - no expensive every_country loop
					add_to_variable = { vp@FROM = state_vp }
					subtract_from_variable = { FROM.vp@ROOT = state_vp }
					clamp_variable = { var = FROM.vp@ROOT min = 0 }
				}
				else = {
					# OPT: Third-party capture — simplified VP assignment
					# Instead of expensive double every_enemy_country loop,
					# give all VP to ROOT (the new controller). This is a good
					# approximation since ROOT is the one who captured it.
					# The old O(N^2) loop was the #1 MP desync source.
					if = {
						limit = { ROOT = { has_war_with = FROM } }
						add_to_variable = { ROOT.vp@FROM = state_vp }
						subtract_from_variable = { FROM.vp@ROOT = state_vp }
						clamp_variable = { var = FROM.vp@ROOT min = 0 }
					}
				}
			}
		}
	}
}
