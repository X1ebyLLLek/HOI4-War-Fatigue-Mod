on_actions = {
	on_startup = {
		effect = {
			random_country = {
				limit = { 
					is_ai = no 
					is_dynamic_country = no 
				}
				country_event = { id = plus_cs.3 }
			}
			every_country = {
				set_variable = { plus_cs_pop_check_timer = 10 }
				set_variable = { plus_cs_need_recalc = 1 }
				set_variable = { core_pop_calc = 0 }
				
				# Инициализация переменных
				# FIX: num_battalions нельзя записать в переменную. Заменено на num_divisions.
				set_variable = { plus_cs_prev_battalions = num_divisions }
				set_variable = { plus_cs_prev_casualties = casualties }
				set_variable = { economy_overheat_counter = 0 }
				set_variable = { plus_cs_prev_econ_law = 0 }
				
				# Новый таймер для пата
				set_variable = { plus_cs_stalemate_timer = 0 }
				
				# Start Optimization Loop
				country_event = { id = plus_cs.999 days = 1 } 
			}
		}
	}

	on_war_relation_added = {
		effect = {
			set_variable = { plus_cs_war_count = 0 }
			every_enemy_country = {
				add_to_variable = { plus_cs_war_count = 1 }
			}
			
			# Инициализация переменных для новых войн
			set_variable = { vp@FROM = 0 }
			set_variable = { FROM.vp@ROOT = 0 }
			set_variable = { plus_cs_last_month_vp = 0 }
			set_variable = { plus_cs_stalemate_timer = 0 }
			
			set_variable = { plus_cs_need_recalc = 1 }
			set_variable = { economy_overheat_counter = 0 }
		}
	}

	on_capitulation = {
		effect = {
			if = {
				limit = { is_major = yes }
				add_threat = -5
			}
			else_if = { 
				limit = { has_offensive_war = yes }
				add_threat = -1
			}
		}
	}

	on_peace = {
		effect = {
			if = {
				limit = { is_major = yes }
				add_threat = -3
			}
			else = {
				add_threat = -0.5
			}

			if = {
				limit = {
					threat > 0.2
					NOT = { 
						any_country = {
							is_major = yes
							has_war = yes
						}
					} 
				}
				add_threat = -1
			}

			set_variable = { plus_cs_last_month_vp = 0 }
			set_variable = { plus_cs_stalemate_timer = 0 }

			if = {
				limit = { has_war = no }
				
				# Очистка флагов
				clr_country_flag = plus_cs_urgent_peace
				clr_country_flag = plus_cs_exhausted
				clr_country_flag = plus_cs_stalemate
				clr_country_flag = plus_cs_war_strain
				clr_country_flag = civil_war_risk_active
				clr_country_flag = plus_cs_revolter_processed

				# Удаление идей 
				remove_ideas = internal_crisis_low
				remove_ideas = internal_crisis_medium
				remove_ideas = internal_crisis_high
				remove_ideas = desperate_defense_effort
				remove_ideas = fight_for_survival
				remove_ideas = panic_retreat_shock

				# Очистка модификаторов
				remove_dynamic_modifier = { modifier = war_inflation_crisis }
				remove_dynamic_modifier = { modifier = manpower_strain_mild }
				remove_dynamic_modifier = { modifier = manpower_strain_high }
				remove_dynamic_modifier = { modifier = manpower_strain_collapse }
				remove_dynamic_modifier = { modifier = officer_shortage_mild }
				remove_dynamic_modifier = { modifier = officer_shortage_high }
				remove_dynamic_modifier = { modifier = officer_shortage_collapse }
				
				# Очистка шока экономики
				remove_dynamic_modifier = { modifier = economic_retooling_shock_minor }
				remove_dynamic_modifier = { modifier = economic_retooling_shock_medium }
				remove_dynamic_modifier = { modifier = economic_retooling_shock_major }
			}
		}
	}

	on_monthly = {
		effect = {
			
			# --- Пассивное остывание мира ---
			if = {
				limit = {
					tag = GER 
					threat > 0.05
				}
				
				set_temp_variable = { world_is_burning = 0 }
				
				if = {
					limit = {
						any_country = {
							is_major = yes
							has_war = yes
						}
					}
					set_temp_variable = { world_is_burning = 1 }
				}
				
				if = {
					limit = { check_variable = { world_is_burning = 0 } }
					add_threat = -0.002
				}
			}

			# ==========================================================
			# MOVED TO TICK SPREADING (plus_cs_optimization.txt)
			# ==========================================================
			# Logic for Inflation, Officer Chaos, Economy Inertia, and Manpower Crisis
			# has been moved to the distributed loop to prevent 1st-of-month lag.


			# ==========================================================
			# 5. ПОЛИТИКА (РАСЧЕТ СИТУАЦИИ)
			# ==========================================================
			if = {
				limit = { has_war = yes }
				
				# --- ЛОГИКА ПАНИКИ (Urgent Peace) ---
				if = {
					limit = {
						OR = {
							has_stability < 0.2
							has_war_support < 0.2
						}
						surrender_progress > 0.10 # Мы не должны паниковать, если побеждаем
						NOT = { has_country_flag = plus_cs_urgent_peace }
					}
					set_country_flag = { flag = plus_cs_urgent_peace days = 180 }
				}
				
				# --- ЛОГИКА ПАТОВОЙ СИТУАЦИИ (Stalemate Logic v2.0) ---
				set_temp_variable = { current_vp_sum = 0 }
				every_enemy_country = {
					add_to_temp_variable = { current_vp_sum = vp@THIS }
				}
				
				if = {
					limit = { check_variable = { plus_cs_last_month_vp > 0 } }
					
					set_temp_variable = { vp_change = current_vp_sum }
					subtract_from_temp_variable = { vp_change = plus_cs_last_month_vp }
					
					# Берем модуль изменения (абсолютное значение)
					if = {
						limit = { check_variable = { vp_change < 0 } }
						multiply_temp_variable = { vp_change = -1 }
					}
					
					# Порог "активности" фронта. (Было 5, стало 15)
					# 15 - это примерно 1-2 средних города или 5-10 провинций.
					# Если изменений меньше этого, считаем ситуацию застойной.
					if = {
						limit = {
							check_variable = { vp_change < 15 } 
						}
						# Увеличиваем счетчик застоя
						add_to_variable = { plus_cs_stalemate_timer = 1 }
					}
					else = {
						# Фронт движется! Сбрасываем таймер
						set_variable = { plus_cs_stalemate_timer = 0 }
						clr_country_flag = plus_cs_stalemate
					}
					
					# Если таймер накопил 6 месяцев (было 4) - включаем флаг пата
					if = {
						limit = { check_variable = { plus_cs_stalemate_timer > 5 } }
						set_country_flag = { flag = plus_cs_stalemate days = 65 } # Даем 2 месяца на решение
					}
				}
				set_variable = { plus_cs_last_month_vp = current_vp_sum }

				if = {
					limit = { casualties > 50000 }
					random_list = {
						2 = { country_event = { id = plus_cs.1001 } }
						98 = { }
					}
				}
				if = {
					limit = { casualties > 100000 }
					random_list = {
						1 = { country_event = { id = plus_cs.1002 } }
						99 = { }
					}
				}

				if = {
					limit = { OR = { has_war_support < 0.2 has_stability < 0.2 } }
					add_timed_idea = { idea = internal_crisis_high days = 60 }
					remove_ideas = internal_crisis_medium
					remove_ideas = internal_crisis_low
				}
				else_if = {
					limit = { OR = { has_war_support < 0.4 has_stability < 0.4 } }
					add_timed_idea = { idea = internal_crisis_medium days = 60 }
					remove_ideas = internal_crisis_high
					remove_ideas = internal_crisis_low
				}
				else_if = {
					limit = { OR = { has_war_support < 0.6 has_stability < 0.6 } }
					add_timed_idea = { idea = internal_crisis_low days = 60 }
					remove_ideas = internal_crisis_high
					remove_ideas = internal_crisis_medium
				}
				else = {
					remove_ideas = internal_crisis_high
					remove_ideas = internal_crisis_medium
					remove_ideas = internal_crisis_low
				}

				if = {
					limit = { 
						has_idea = internal_crisis_high 
						NOT = { AND = { date < 1938.1.1 has_civil_war = yes } }
					}
					if = {
						limit = { has_government = democratic }
						random_list = {
							40 = { add_popularity = { ideology = fascism popularity = 0.1 } }
							40 = { add_popularity = { ideology = communism popularity = 0.1 } }
							20 = { add_popularity = { ideology = neutrality popularity = 0.1 } }
						}
					}
					else_if = {
						limit = { has_government = neutrality }
						random_list = {
							35 = { add_popularity = { ideology = fascism popularity = 0.1 } } 
							35 = { add_popularity = { ideology = communism popularity = 0.1 } } 
							30 = { add_popularity = { ideology = democratic popularity = 0.1 } }
						}
					}
					else_if = {
						limit = { has_government = fascism }
						random_list = {
							50 = { add_popularity = { ideology = communism popularity = 0.1 } }
							25 = { add_popularity = { ideology = democratic popularity = 0.05 } add_popularity = { ideology = neutrality popularity = 0.05 } }
							25 = { add_popularity = { ideology = neutrality popularity = 0.1 } }
						}
					}
					else_if = {
						limit = { has_government = communism }
						random_list = {
							50 = { add_popularity = { ideology = fascism popularity = 0.1 } }
							25 = { add_popularity = { ideology = democratic popularity = 0.05 } add_popularity = { ideology = neutrality popularity = 0.05 } }
							25 = { add_popularity = { ideology = neutrality popularity = 0.1 } }
					  }
					}
				}
			}
		}
	}

	# --- OPTIMIZATION HOOKS (Tick-Spreading) ---
	
	on_daily = {
		effect = {
			# Fail-safe: Если страна "выпала" из цикла или появилась только что
			# OPTIMIZATION: Check only 10% of the time to reduce overhead
			random_list = {
				10 = {
					random_country = {
						limit = {
							NOT = { has_country_flag = plus_cs_loop_active }
							exists = yes
						}
						country_event = { id = plus_cs.999 }
					}
				}
				90 = { }
			}
		}
	}

	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					FROM.FROM = { state_strategic_value > 15 is_core_of = ROOT }
					ROOT = { has_war = yes }
				}
				ROOT = {
					remove_ideas = panic_retreat_shock
					add_timed_idea = { idea = panic_retreat_shock days = 10 }
				}
			}
			
			if = {
				limit = {
					FROM.FROM = { is_core_of = FROM }
					FROM = { has_idea = fight_for_survival }
					ROOT = { has_war_with = FROM }
				}
				FROM.FROM = {
					damage_building = { type = infrastructure damage = 1.0 }
					damage_building = { type = supply_node damage = 1.0 }
					damage_building = { type = rail_way damage = 1.0 }
					random_list = {
						50 = { damage_building = { type = arms_factory damage = 0.5 } damage_building = { type = industrial_complex damage = 0.5 } }
						50 = { }
					}
				}
			}

			if = { limit = { FROM.FROM = { is_core_of = ROOT } } set_variable = { plus_cs_need_recalc = 1 } }
			if = { limit = { FROM.FROM = { is_core_of = FROM } } FROM = { set_variable = { plus_cs_need_recalc = 1 } } }

			# --- ОПТИМИЗАЦИЯ: НЕ СЧИТАЕМ, ЕСЛИ ПРОВИНЦИЯ ПУСТАЯ ---
			if = {
				limit = { FROM.FROM = { state_strategic_value > 0 } }
				
				set_temp_variable = { state_vp = FROM.FROM.state_strategic_value }
				set_temp_variable = { tot_pts = 0 }
				
				# OPTIMIZATION: Merged double every_country loop into one
				# First pass: calculate total points
				set_temp_variable = { tot_pts = 0 }
				every_country = {
					limit = { 
						has_war_with = FROM 
						# OPTIMIZATION: Skip if no armies here
						check_variable = { num_armies_in_state@FROM.FROM > 0 }
					}
					add_to_temp_variable = { tot_pts = num_armies_in_state@FROM.FROM }
					
					# Store this country's contribution for second pass
					set_temp_variable = { my_pts@THIS = num_armies_in_state@FROM.FROM }
				}
				
				# Second pass: distribute VP proportionally (only if there were armies)
				if = {
					limit = { check_variable = { tot_pts > 0 } }
					every_country = {
						limit = { 
							has_war_with = FROM 
							check_variable = { my_pts@THIS > 0 }
						}
						set_temp_variable = { pts = my_pts@THIS }
						divide_temp_variable = { pts = tot_pts }
						multiply_temp_variable = { pts = state_vp }
						add_to_variable = { vp@FROM = pts }
						subtract_from_variable = { FROM.vp@THIS = pts }
						clamp_variable = { var = FROM.vp@THIS min = 0 }
					}
				}
			}
		}
	}
}
