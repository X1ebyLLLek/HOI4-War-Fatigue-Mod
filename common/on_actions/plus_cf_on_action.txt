on_actions = {
	on_startup = {
		effect = {
			random_country = {
				limit = { 
					is_ai = no 
					is_dynamic_country = no 
				}
				country_event = { id = plus_cs.3 }
			}
			every_country = {
				set_variable = { plus_cs_pop_check_timer = 10 }
				set_variable = { plus_cs_need_recalc = 1 }
				set_variable = { core_pop_calc = 0 }
				
				# Инициализация переменных
				# FIX: num_battalions нельзя записать в переменную. Заменено на num_divisions.
				set_variable = { plus_cs_prev_battalions = num_divisions }
				set_variable = { plus_cs_prev_casualties = casualties }
				set_variable = { economy_overheat_counter = 0 }
				set_variable = { plus_cs_prev_econ_law = 0 }
				
				# Новый таймер для пата
				set_variable = { plus_cs_stalemate_timer = 0 }
				
				# Start Optimization Loop
				country_event = { id = plus_cs.999 days = 1 } 
			}
		}
	}

	on_war_relation_added = {
		effect = {
			set_variable = { plus_cs_war_count = 0 }
			every_enemy_country = {
				add_to_variable = { plus_cs_war_count = 1 }
			}
			
			# Инициализация переменных для новых войн
			set_variable = { vp@FROM = 0 }
			set_variable = { FROM.vp@ROOT = 0 }
			set_variable = { plus_cs_last_month_vp = 0 }
			set_variable = { plus_cs_stalemate_timer = 0 }
			
			set_variable = { plus_cs_need_recalc = 1 }
			set_variable = { economy_overheat_counter = 0 }
		}
	}

	on_capitulation = {
		effect = {
			if = {
				limit = { is_major = yes }
				add_threat = -5
			}
			else_if = { 
				limit = { has_offensive_war = yes }
				add_threat = -1
			}
		}
	}

	on_peace = {
		effect = {
			if = {
				limit = { is_major = yes }
				add_threat = -3
			}
			else = {
				add_threat = -0.5
			}

			if = {
				limit = {
					threat > 0.2
					NOT = { 
						any_country = {
							is_major = yes
							has_war = yes
						}
					} 
				}
				add_threat = -1
			}

			set_variable = { plus_cs_last_month_vp = 0 }
			set_variable = { plus_cs_stalemate_timer = 0 }

			if = {
				limit = { has_war = no }
				
				# Очистка флагов
				clr_country_flag = plus_cs_urgent_peace
				clr_country_flag = plus_cs_exhausted
				clr_country_flag = plus_cs_stalemate
				clr_country_flag = plus_cs_war_strain
				clr_country_flag = civil_war_risk_active
				clr_country_flag = plus_cs_revolter_processed

				# Удаление идей 
				remove_ideas = internal_crisis_low
				remove_ideas = internal_crisis_medium
				remove_ideas = internal_crisis_high
				remove_ideas = desperate_defense_effort
				remove_ideas = fight_for_survival
				remove_ideas = panic_retreat_shock

				# Очистка модификаторов
				remove_dynamic_modifier = { modifier = war_inflation_crisis }
				remove_dynamic_modifier = { modifier = manpower_strain_mild }
				remove_dynamic_modifier = { modifier = manpower_strain_high }
				remove_dynamic_modifier = { modifier = manpower_strain_collapse }
				remove_dynamic_modifier = { modifier = officer_shortage_mild }
				remove_dynamic_modifier = { modifier = officer_shortage_high }
				remove_dynamic_modifier = { modifier = officer_shortage_collapse }
				
				# Очистка шока экономики
				remove_dynamic_modifier = { modifier = economic_retooling_shock_minor }
				remove_dynamic_modifier = { modifier = economic_retooling_shock_medium }
				remove_dynamic_modifier = { modifier = economic_retooling_shock_major }
			}
		}
	}

	on_monthly = {
		effect = {
			
			# --- Пассивное остывание мира ---
			if = {
				limit = {
					tag = GER 
					threat > 0.05
				}
				
				set_temp_variable = { world_is_burning = 0 }
				
				if = {
					limit = {
						any_country = {
							is_major = yes
							has_war = yes
						}
					}
					set_temp_variable = { world_is_burning = 1 }
				}
				
				if = {
					limit = { check_variable = { world_is_burning = 0 } }
					add_threat = -0.002
				}
			}

			# ==========================================================
			# НОВАЯ МЕХАНИКА ИНФЛЯЦИИ (SMART WAR INFLATION)
			# ==========================================================
			if = {
				limit = {
					has_war = yes
					OR = {
						# 1. ПОЛНЫЙ ПОЛИТИЧЕСКИЙ КОЛЛАПС (Hard Floor)
						# Экономика рушится, так как власти больше нет (< 20%).
						has_stability < 0.20
						
						# 2. ЭКОНОМИЧЕСКАЯ ПАНИКА (Conditional)
						# Стабильность шаткая (20-45%), НО инфляция начнется только при РЕАЛЬНЫХ проблемах.
						AND = {
							has_stability < 0.45
							OR = {
								# Враг захватил >10% наших победных очков (мы проигрываем)
								surrender_progress > 0.10
								
								# Мы потеряли слишком много людей (кризис рекрутов)
								has_dynamic_modifier = manpower_strain_mild
								has_dynamic_modifier = manpower_strain_high
								
								# Население категорически против войны (< 30%)
								has_war_support < 0.30
							}
						}

						# 3. ИСТОЩЕНИЕ РЕСУРСОВ (Hard Trigger)
						# Если работать некому (Heavy/Collapse), экономика падает независимо от стабильности.
						has_dynamic_modifier = manpower_strain_high
						has_dynamic_modifier = manpower_strain_collapse
					}
				}
				# Если условия совпали - вешаем инфляцию на 30 дней (до след. проверки)
				add_dynamic_modifier = { modifier = war_inflation_crisis days = 30 }
				
				# --- НАКОПИТЕЛЬНЫЙ УРОН ОТ ИНФЛЯЦИИ ---
				# Отнимаем по 0.5% каждый месяц, пока висит кризис
				add_stability = -0.005
				add_war_support = -0.005
			}
			else = {
				# Если война кончилась ИЛИ ситуация стабилизировалась - снимаем
				remove_dynamic_modifier = { modifier = war_inflation_crisis }
			}

			# ==========================================================
			# 1. МЕХАНИКА "ХАОС ОФИЦЕРОВ" 
			# ==========================================================
			
			if = {
				limit = { 
					OR = {
						NOT = { check_variable = { plus_cs_prev_battalions > 0 } }
						check_variable = { plus_cs_prev_battalions < 10 }
					}
				}
				# FIX: num_battalions -> num_divisions
				set_variable = { plus_cs_prev_battalions = num_divisions }
				set_variable = { plus_cs_prev_casualties = casualties }
			}
			else = {
				set_temp_variable = { officer_cap_calc = num_research_slots }
				multiply_temp_variable = { officer_cap_calc = 200 }
				add_to_temp_variable = { officer_cap_calc = 400 }
				
				if = {
					limit = { is_major = yes }
					add_to_temp_variable = { officer_cap_calc = 500 }
				}

				# FIX: num_battalions -> num_divisions
				set_temp_variable = { total_strain = num_divisions }
				subtract_from_temp_variable = { total_strain = var:plus_cs_prev_battalions }
				clamp_temp_variable = { var = total_strain min = 0 }
				multiply_temp_variable = { total_strain = 0.5 } 

				set_temp_variable = { casualty_load = casualties }
				subtract_from_temp_variable = { casualty_load = var:plus_cs_prev_casualties }
				clamp_temp_variable = { var = casualty_load min = 0 }
				divide_temp_variable = { casualty_load = 3500 }

				add_to_temp_variable = { total_strain = var:casualty_load }

				if = {
					limit = { num_divisions > 0 }
					divide_temp_variable = { total_strain = num_divisions }
				}

				# Штраф за лимит
				# FIX: num_battalions -> num_divisions
				if = {
					limit = { check_variable = { var = num_divisions value = var:officer_cap_calc compare = greater_than } }
					multiply_temp_variable = { total_strain = 1.2 }
					
					set_temp_variable = { double_cap = var:officer_cap_calc }
					multiply_temp_variable = { double_cap = 2 }
					if = {
						limit = { check_variable = { var = num_divisions value = var:double_cap compare = greater_than } }
						multiply_temp_variable = { total_strain = 1.66 } 
					}
				}

				# Применение модификаторов 
				if = {
					limit = { check_variable = { total_strain > 0.40 } }
					add_dynamic_modifier = { modifier = officer_shortage_collapse days = 60 }
					remove_dynamic_modifier = { modifier = officer_shortage_high }
					remove_dynamic_modifier = { modifier = officer_shortage_mild }
				}
				else_if = {
					limit = { check_variable = { total_strain > 0.20 } }
					add_dynamic_modifier = { modifier = officer_shortage_high days = 60 }
					remove_dynamic_modifier = { modifier = officer_shortage_collapse }
					remove_dynamic_modifier = { modifier = officer_shortage_mild }
				}
				else_if = {
					limit = { check_variable = { total_strain > 0.08 } }
					add_dynamic_modifier = { modifier = officer_shortage_mild days = 60 }
					remove_dynamic_modifier = { modifier = officer_shortage_collapse }
					remove_dynamic_modifier = { modifier = officer_shortage_high }
				}
				else = {
					remove_dynamic_modifier = { modifier = officer_shortage_collapse }
					remove_dynamic_modifier = { modifier = officer_shortage_high }
					remove_dynamic_modifier = { modifier = officer_shortage_mild }
				}

				set_variable = { plus_cs_prev_battalions = num_divisions }
				set_variable = { plus_cs_prev_casualties = casualties }
			} 

			# ==========================================================
			# 2. ИНЕРЦИЯ ЭКОНОМИКИ
			# ==========================================================
			set_temp_variable = { current_econ_law_idx = 0 }
			if = { limit = { has_idea = civilian_economy } set_temp_variable = { current_econ_law_idx = 1 } }
			else_if = { limit = { has_idea = low_economic_mobilisation } set_temp_variable = { current_econ_law_idx = 2 } }
			else_if = { limit = { has_idea = partial_economic_mobilisation } set_temp_variable = { current_econ_law_idx = 3 } }
			else_if = { limit = { has_idea = war_economy } set_temp_variable = { current_econ_law_idx = 4 } }
			else_if = { limit = { has_idea = tot_economic_mobilisation } set_temp_variable = { current_econ_law_idx = 5 } }

			if = {
				limit = { NOT = { check_variable = { plus_cs_prev_econ_law > 0 } } }
				set_variable = { plus_cs_prev_econ_law = current_econ_law_idx }
			}
			else_if = {
				limit = { 
					check_variable = { current_econ_law_idx > 0 }
					NOT = { check_variable = { plus_cs_prev_econ_law = current_econ_law_idx } } 
				}

				set_temp_variable = { law_diff = current_econ_law_idx }
				subtract_from_temp_variable = { law_diff = plus_cs_prev_econ_law }
				
				if = {
					limit = { check_variable = { law_diff < 0 } }
					multiply_temp_variable = { law_diff = -1 }
				}

				remove_dynamic_modifier = { modifier = economic_retooling_shock_minor }
				remove_dynamic_modifier = { modifier = economic_retooling_shock_medium }
				remove_dynamic_modifier = { modifier = economic_retooling_shock_major }

				if = {
					limit = { check_variable = { law_diff > 2 } }
					add_dynamic_modifier = { modifier = economic_retooling_shock_major days = 120 }
				}
				else_if = {
					limit = { check_variable = { law_diff = 2 } }
					add_dynamic_modifier = { modifier = economic_retooling_shock_medium days = 90 }
				}
				else = {
					add_dynamic_modifier = { modifier = economic_retooling_shock_minor days = 60 }
				}
				set_variable = { plus_cs_prev_econ_law = current_econ_law_idx }
			}

			# ==========================================================
			# 3. ТЕНЕВАЯ ЭКОНОМИКА
			# ==========================================================
			
			remove_dynamic_modifier = { modifier = shadow_econ_civilian }
			remove_dynamic_modifier = { modifier = shadow_econ_early }
			remove_dynamic_modifier = { modifier = shadow_econ_partial }
			remove_dynamic_modifier = { modifier = shadow_econ_war }
			remove_dynamic_modifier = { modifier = shadow_econ_total }

			if = {
				limit = { has_idea = civilian_economy }
				add_dynamic_modifier = { modifier = shadow_econ_civilian days = 30 }
			}
			else_if = {
				limit = { has_idea = low_economic_mobilisation } 
				add_dynamic_modifier = { modifier = shadow_econ_early days = 30 }
			}
			else_if = {
				limit = { has_idea = partial_economic_mobilisation }
				add_dynamic_modifier = { modifier = shadow_econ_partial days = 30 }
			}
			else_if = {
				limit = { has_idea = war_economy }
				add_dynamic_modifier = { modifier = shadow_econ_war days = 30 }
			}
			else_if = {
				limit = { has_idea = tot_economic_mobilisation }
				add_dynamic_modifier = { modifier = shadow_econ_total days = 30 }
			}

			if = {
				limit = {
					has_war = no
					OR = {
						has_idea = war_economy
						has_idea = tot_economic_mobilisation
					}
				}
				add_to_variable = { economy_overheat_counter = 1 }
				
				if = {
					limit = { check_variable = { economy_overheat_counter > 6 } }
					add_political_power = -10
					add_stability = -0.01
				}
			}
			else = {
				set_variable = { economy_overheat_counter = 0 }
			}

			if = {
				limit = { has_idea = service_by_requirement }
				add_dynamic_modifier = { modifier = shadow_law_service_penalty days = 30 }
			}
			else_if = {
				limit = { has_idea = all_adults_serve }
				add_dynamic_modifier = { modifier = shadow_law_adults_penalty days = 30 }
			}
			else_if = {
				limit = { has_idea = scraping_the_barrel }
				add_dynamic_modifier = { modifier = shadow_law_scraping_penalty days = 30 }
			}

			# ==========================================================
			# 4. КРИЗИС ЖИВОЙ СИЛЫ
			# ==========================================================
			if = {
				limit = { has_war = yes }

				add_to_variable = { plus_cs_pop_check_timer = 1 }

				if = {
					limit = {
						OR = {
							check_variable = { plus_cs_pop_check_timer > 3 } 
							check_variable = { plus_cs_need_recalc = 1 }
							check_variable = { core_pop_calc = 0 }
						}
					}
					set_variable = { plus_cs_pop_check_timer = 0 }
					set_variable = { plus_cs_need_recalc = 0 }
					set_variable = { core_pop_calc = 0 }
					
					every_owned_state = {
						if = {
							limit = { is_core_of = ROOT }
							add_to_temp_variable = { core_pop_calc = state_population }
						}
						else = {
							set_temp_variable = { local_pop = state_population }
							multiply_temp_variable = { local_pop = 0.15 }
							add_to_temp_variable = { core_pop_calc = local_pop }
						}
					}
					every_subject_country = {
						set_temp_variable = { subj_pop = core_population }
						multiply_temp_variable = { subj_pop = 0.25 }
						add_to_temp_variable = { ROOT.core_pop_calc = subj_pop }
					}
				}

				# FIX: num_battalions -> num_divisions
				set_temp_variable = { total_drain = num_divisions }
				multiply_temp_variable = { total_drain = 1000 }
				add_to_temp_variable = { total_drain = casualties }
				
				set_temp_variable = { limit_base = var:core_pop_calc }
				
				# Collapse
				set_temp_variable = { limit_check = var:limit_base }
				multiply_temp_variable = { limit_check = 0.10 }
				add_to_temp_variable = { limit_check = 200000 }

				if = {
					limit = { check_variable = { var = total_drain value = var:limit_check compare = greater_than } }
					add_dynamic_modifier = { modifier = manpower_strain_collapse days = 30 }
					remove_dynamic_modifier = { modifier = manpower_strain_high }
					remove_dynamic_modifier = { modifier = manpower_strain_mild }
					set_country_flag = { flag = plus_cs_exhausted days = 30 }
					clr_country_flag = plus_cs_war_strain
				}
				else = {
					# High
					set_temp_variable = { limit_check = var:limit_base }
					multiply_temp_variable = { limit_check = 0.05 }
					add_to_temp_variable = { limit_check = 125000 }
					
					if = {
						limit = { check_variable = { var = total_drain value = var:limit_check compare = greater_than } }
						add_dynamic_modifier = { modifier = manpower_strain_high days = 30 }
						remove_dynamic_modifier = { modifier = manpower_strain_collapse }
						remove_dynamic_modifier = { modifier = manpower_strain_mild }
						set_country_flag = { flag = plus_cs_exhausted days = 30 }
						clr_country_flag = plus_cs_war_strain
					}
					else = {
						# Mild
						set_temp_variable = { limit_check = var:limit_base }
						multiply_temp_variable = { limit_check = 0.02 }
						add_to_temp_variable = { limit_check = 75000 }
						
						if = {
							limit = { check_variable = { var = total_drain value = var:limit_check compare = greater_than } }
							add_dynamic_modifier = { modifier = manpower_strain_mild days = 30 }
							remove_dynamic_modifier = { modifier = manpower_strain_collapse }
							remove_dynamic_modifier = { modifier = manpower_strain_high }
							if = { limit = { NOT = { has_country_flag = plus_cs_exhausted } } set_country_flag = { flag = plus_cs_war_strain days = 30 } }
						}
						else = {
							remove_dynamic_modifier = { modifier = manpower_strain_collapse }
							remove_dynamic_modifier = { modifier = manpower_strain_high }
							remove_dynamic_modifier = { modifier = manpower_strain_mild }
						}
					}
				}

				if = {
					limit = { 
						is_major = no
						has_war_with_major = yes
						any_owned_state = {
							is_core_of = ROOT
							is_controlled_by = ROOT
							check_variable = { ROOT.surrender_progress > 0.05 }
						}
					}
					add_timed_idea = { idea = desperate_defense_effort days = 30 }
				}
				else = {
					remove_ideas = desperate_defense_effort
				}
			}
			else = {
				remove_dynamic_modifier = { modifier = manpower_strain_mild }
				remove_dynamic_modifier = { modifier = manpower_strain_high }
				remove_dynamic_modifier = { modifier = manpower_strain_collapse }
				remove_ideas = desperate_defense_effort
			}

			# ==========================================================
			# 5. ПОЛИТИКА (РАСЧЕТ СИТУАЦИИ)
			# ==========================================================
			if = {
				limit = { has_war = yes }
				
				# --- ЛОГИКА ПАНИКИ (Urgent Peace) ---
				if = {
					limit = {
						OR = {
							has_stability < 0.2
							has_war_support < 0.2
						}
						surrender_progress > 0.10 # Мы не должны паниковать, если побеждаем
						NOT = { has_country_flag = plus_cs_urgent_peace }
					}
					set_country_flag = { flag = plus_cs_urgent_peace days = 180 }
				}
				
				# --- ЛОГИКА ПАТОВОЙ СИТУАЦИИ (Stalemate Logic v2.0) ---
				set_temp_variable = { current_vp_sum = 0 }
				every_enemy_country = {
					add_to_temp_variable = { current_vp_sum = vp@THIS }
				}
				
				if = {
					limit = { check_variable = { plus_cs_last_month_vp > 0 } }
					
					set_temp_variable = { vp_change = current_vp_sum }
					subtract_from_temp_variable = { vp_change = plus_cs_last_month_vp }
					
					# Берем модуль изменения (абсолютное значение)
					if = {
						limit = { check_variable = { vp_change < 0 } }
						multiply_temp_variable = { vp_change = -1 }
					}
					
					# Порог "активности" фронта. (Было 5, стало 15)
					# 15 - это примерно 1-2 средних города или 5-10 провинций.
					# Если изменений меньше этого, считаем ситуацию застойной.
					if = {
						limit = {
							check_variable = { vp_change < 15 } 
						}
						# Увеличиваем счетчик застоя
						add_to_variable = { plus_cs_stalemate_timer = 1 }
					}
					else = {
						# Фронт движется! Сбрасываем таймер
						set_variable = { plus_cs_stalemate_timer = 0 }
						clr_country_flag = plus_cs_stalemate
					}
					
					# Если таймер накопил 6 месяцев (было 4) - включаем флаг пата
					if = {
						limit = { check_variable = { plus_cs_stalemate_timer > 5 } }
						set_country_flag = { flag = plus_cs_stalemate days = 65 } # Даем 2 месяца на решение
					}
				}
				set_variable = { plus_cs_last_month_vp = current_vp_sum }

				if = {
					limit = { casualties > 50000 }
					random_list = {
						2 = { country_event = { id = plus_cs.1001 } }
						98 = { }
					}
				}
				if = {
					limit = { casualties > 100000 }
					random_list = {
						1 = { country_event = { id = plus_cs.1002 } }
						99 = { }
					}
				}

				if = {
					limit = { OR = { has_war_support < 0.2 has_stability < 0.2 } }
					add_timed_idea = { idea = internal_crisis_high days = 60 }
					remove_ideas = internal_crisis_medium
					remove_ideas = internal_crisis_low
				}
				else_if = {
					limit = { OR = { has_war_support < 0.4 has_stability < 0.4 } }
					add_timed_idea = { idea = internal_crisis_medium days = 60 }
					remove_ideas = internal_crisis_high
					remove_ideas = internal_crisis_low
				}
				else_if = {
					limit = { OR = { has_war_support < 0.6 has_stability < 0.6 } }
					add_timed_idea = { idea = internal_crisis_low days = 60 }
					remove_ideas = internal_crisis_high
					remove_ideas = internal_crisis_medium
				}
				else = {
					remove_ideas = internal_crisis_high
					remove_ideas = internal_crisis_medium
					remove_ideas = internal_crisis_low
				}

				if = {
					limit = { 
						has_idea = internal_crisis_high 
						NOT = { AND = { date < 1938.1.1 has_civil_war = yes } }
					}
					if = {
						limit = { has_government = democratic }
						random_list = {
							40 = { add_popularity = { ideology = fascism popularity = 0.1 } }
							40 = { add_popularity = { ideology = communism popularity = 0.1 } }
							20 = { add_popularity = { ideology = neutrality popularity = 0.1 } }
						}
					}
					else_if = {
						limit = { has_government = neutrality }
						random_list = {
							35 = { add_popularity = { ideology = fascism popularity = 0.1 } } 
							35 = { add_popularity = { ideology = communism popularity = 0.1 } } 
							30 = { add_popularity = { ideology = democratic popularity = 0.1 } }
						}
					}
					else_if = {
						limit = { has_government = fascism }
						random_list = {
							50 = { add_popularity = { ideology = communism popularity = 0.1 } }
							25 = { add_popularity = { ideology = democratic popularity = 0.05 } add_popularity = { ideology = neutrality popularity = 0.05 } }
							25 = { add_popularity = { ideology = neutrality popularity = 0.1 } }
						}
					}
					else_if = {
						limit = { has_government = communism }
						random_list = {
							50 = { add_popularity = { ideology = fascism popularity = 0.1 } }
							25 = { add_popularity = { ideology = democratic popularity = 0.05 } add_popularity = { ideology = neutrality popularity = 0.05 } }
							25 = { add_popularity = { ideology = neutrality popularity = 0.1 } }
					  }
					}
				}
			}
		}
	}

	# --- OPTIMIZATION HOOKS (Tick-Spreading) ---
	
	on_daily = {
		effect = {
			# Fail-safe: Если страна "выпала" из цикла или появилась только что
			random_country = {
				limit = {
					NOT = { has_country_flag = plus_cs_loop_active }
				}
				country_event = { id = plus_cs.999 }
			}
		}
	}

	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					FROM.FROM = { state_strategic_value > 15 is_core_of = ROOT }
					ROOT = { has_war = yes }
				}
				ROOT = {
					remove_ideas = panic_retreat_shock
					add_timed_idea = { idea = panic_retreat_shock days = 10 }
				}
			}
			
			if = {
				limit = {
					FROM.FROM = { is_core_of = FROM }
					FROM = { has_idea = fight_for_survival }
					ROOT = { has_war_with = FROM }
				}
				FROM.FROM = {
					damage_building = { type = infrastructure damage = 1.0 }
					damage_building = { type = supply_node damage = 1.0 }
					damage_building = { type = rail_way damage = 1.0 }
					random_list = {
						50 = { damage_building = { type = arms_factory damage = 0.5 } damage_building = { type = industrial_complex damage = 0.5 } }
						50 = { }
					}
				}
			}

			if = { limit = { FROM.FROM = { is_core_of = ROOT } } set_variable = { plus_cs_need_recalc = 1 } }
			if = { limit = { FROM.FROM = { is_core_of = FROM } } FROM = { set_variable = { plus_cs_need_recalc = 1 } } }

			# --- ОПТИМИЗАЦИЯ: НЕ СЧИТАЕМ, ЕСЛИ ПРОВИНЦИЯ ПУСТАЯ ---
			if = {
				limit = { FROM.FROM = { state_strategic_value > 0 } }
				
				set_temp_variable = { state_vp = FROM.FROM.state_strategic_value }
				set_temp_variable = { tot_pts = 0 }
				
				every_country = {
					limit = { has_war_with = FROM }
					add_to_temp_variable = { tot_pts = num_armies_in_state@FROM.FROM }
				}
				
				every_country = {
					limit = { has_war_with = FROM }
					set_temp_variable = { pts = num_armies_in_state@FROM.FROM }
					if = {
						limit = { check_variable = { tot_pts > 0 } }
						divide_temp_variable = { pts = tot_pts }
					}
					multiply_temp_variable = { pts = state_vp }
					add_to_variable = { vp@FROM = pts }
					subtract_from_variable = { FROM.vp@THIS = pts }
					clamp_variable = { var = FROM.vp@THIS min = 0 }
				}
			}
		}
	}
