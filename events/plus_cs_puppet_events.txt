# Event 210: Puppet Fate (Subject Transfer)
country_event = {
	id = plus_cs.210
	title = plus_cs.puppet_fate.t
	desc = plus_cs.puppet_fate.d
	picture = GFX_report_event_generic_sign_treaty1
	is_triggered_only = yes

	option = {
		name = plus_cs.puppet_fate.a
		ai_chance = {
			base = 80
			modifier = { factor = 0.5 has_government = fascism }
			modifier = { factor = 0.5 has_government = communism }
			# Desperation
			modifier = {
				factor = 5.0
				OR = {
					has_idea = political_paralysis
					has_dynamic_modifier = homefront_collapse
					has_dynamic_modifier = total_exhaustion
				}
			}
		}

		if = {
			limit = { has_war_with = event_target:new_potential_master }
			white_peace = event_target:new_potential_master
		}
		
		event_target:new_potential_master = {
			puppet = ROOT
		}
		
		add_timed_idea = {
			idea = high_status_puppet_benefit
			days = 730
		}
		
		add_stability = 0.10
	}

	option = {
		name = plus_cs.puppet_fate.b
		ai_chance = {
			base = 20
			modifier = { factor = 2.0 has_government = fascism }
			modifier = { factor = 1.5 strength_ratio = { tag = event_target:new_potential_master ratio > 0.4 } }
			# Refuse to submit to a dying empire
			modifier = {
				factor = 100.0
				event_target:new_potential_master = {
					OR = {
						has_idea = political_paralysis
						has_idea = internal_crisis_high
					}
				}
			}
		}

		set_autonomy = {
			target = FROM
			autonomy_state = autonomy_free
		}
		
		add_war_support = 0.15
		add_stability = -0.05
		add_timed_idea = { idea = revanchism_spirit days = 365 }
	}
}

# --- ИСПРАВЛЕННЫЙ ИВЕНТ 300: Выбор покровителя ---
country_event = {
	id = plus_cs.300
	title = plus_cs.300.t
	desc = plus_cs.300.d
	picture = GFX_report_event_generic_sign_treaty1
	is_triggered_only = yes

	immediate = {
		# 1. Находим, с кем воюют революционеры (Гражданская война)
		if = {
			limit = {
				any_enemy_country = {
					original_tag = ROOT
				}
			}
			random_enemy_country = {
				limit = {
					original_tag = ROOT
				}
				save_event_target_as = cw_target
			}
		}
		
		# 2. Ищем лучшего кандидата среди врагов цели
		# КОМПОЗИТНЫЙ СКОРИНГ: VP + потери + соседство + мажор
		if = {
			limit = { has_event_target = cw_target }
			
			# Переменные для сравнения
			set_temp_variable = { best_score = -1 }
			
			event_target:cw_target = {
				every_enemy_country = {
					limit = {
						NOT = { tag = ROOT }          # Не революционеры
						NOT = { is_puppet_of = ROOT } # Не марионетки революционеров
						NOT = { is_puppet = yes }     # Ищем независимого покровителя
					}
					
					# === COMPOSITE SCORE ===
					# 1. VP захваченных территорий (основной показатель)
					set_temp_variable = { current_score = var:vp@event_target:cw_target }
					if = {
						limit = { NOT = { check_variable = { current_score > -1 } } }
						set_temp_variable = { current_score = 0 }
					}
					
					# 2. Потери в войне (показывает реальное участие)
					# casualties = общие потери страны. Делим на 1000, cap 50
					set_temp_variable = { cas_bonus = casualties }
					divide_temp_variable = { cas_bonus = 1000 }
					clamp_temp_variable = { var = cas_bonus max = 50 }
					add_to_temp_variable = { current_score = cas_bonus }
					
					# 3. Соседство с целью (сухопутная граница = реальная угроза)
					if = {
						limit = { is_neighbor_of = event_target:cw_target }
						add_to_temp_variable = { current_score = 30 }
					}
					
					# 4. Мажор-бонус (небольшой, чтобы не перебивал вклад)
					if = {
						limit = { is_major = yes }
						add_to_temp_variable = { current_score = 10 }
					}
					
					# СРАВНЕНИЕ: Если этот кандидат лучше предыдущего
					if = {
						limit = { check_variable = { current_score > best_score } }
						set_temp_variable = { best_score = current_score }
						save_event_target_as = strong_master # Сохраняем лидера гонки
					}
				}
			}
			
			# 3. ПОСЛЕДНИЙ РЕЗЕРВ (Если никто не набрал > 0 — все только вступили)
			# Берём по фабрикам как тай-брейкер
			if = {
				limit = { NOT = { has_event_target = strong_master } }
				event_target:cw_target = {
					random_enemy_country = {
						limit = { NOT = { tag = ROOT } NOT = { is_puppet = yes } }
						save_event_target_as = strong_master
					}
					every_enemy_country = {
						limit = {
							NOT = { tag = ROOT }
							NOT = { is_puppet = yes }
							num_of_factories > event_target:strong_master.num_of_factories
						}
						save_event_target_as = strong_master
					}
				}
			}
		}
	}

	option = {
		name = plus_cs.300.a
		trigger = { has_event_target = strong_master }
		
		ai_chance = {
			base = 90
			# Если покровитель совсем слабый по сравнению с целью, ИИ может побояться
			modifier = {
				factor = 0.1
				strength_ratio = { tag = event_target:strong_master ratio > 1.5 } # Враг слабее покровителя
			}
			# ИЛИ ЕСЛИ ПОКРОВИТЕЛЬ В КРИЗИСЕ - зачем к нему идти?
			modifier = {
				factor = 0.01
				event_target:strong_master = {
					OR = {
						has_idea = political_paralysis
						has_idea = internal_crisis_high
					}
				}
			}
		}

		# BUGFIX 2b: Give the human player first choice to patron the revolter.
		# If the player is at war with cw_target, they get event 305 instead of the AI strong_master.
		if = {
			limit = {
				has_event_target = cw_target
				event_target:cw_target = {
					any_enemy_country = {
						is_ai = no
						NOT = { tag = ROOT }
						NOT = { is_puppet = yes }
					}
				}
			}
			event_target:cw_target = {
				random_enemy_country = {
					limit = {
						is_ai = no
						NOT = { tag = ROOT }
						NOT = { is_puppet = yes }
					}
					country_event = { id = plus_cs.305 }
				}
			}
		}
		else_if = {
			limit = { has_event_target = strong_master }
			event_target:strong_master = {
				country_event = { id = plus_cs.305 }
			}
		}

		add_stability = 0.10
		add_war_support = 0.10
	}

	option = {
		name = plus_cs.300.b
		trigger = { has_event_target = strong_master }
		ai_chance = {
			base = 10
			modifier = {
				factor = 2.0
				has_government = fascism
			}
			# Если мы сильнее врага, зачем нам хозяин?
			modifier = {
				factor = 5.0
				strength_ratio = { tag = event_target:cw_target ratio < 0.5 }
			}
			# Если мы сами в полном коллапсе, мы не будем объявлять войну случайному мажору
			modifier = {
				factor = 0.0
				OR = {
					has_idea = political_paralysis
					has_dynamic_modifier = homefront_collapse
					has_dynamic_modifier = total_exhaustion
				}
			}
		}

		if = {
			limit = { NOT = { has_war_with = event_target:strong_master } }
			declare_war_on = {
				target = event_target:strong_master
				type = annex_everything
			}
		}
		
		add_stability = 0.05
		add_war_support = 0.15
		add_timed_idea = { idea = revanchism_spirit days = 365 }
	}

	option = {
		name = plus_cs.300.c # Fallback
		trigger = { NOT = { has_event_target = strong_master } }
		# Do nothing
	}
}

country_event = {

	option = {

	}

	# BUGFIX 2b: Player can decline to patron the revolter — they remain independent.
	option = {
		name = plus_cs.305.b
		ai_chance = { base = 0 }
	}
}
