add_namespace = plus_cs



# ==========================================
# WHITE PEACE EVENTS
# ==========================================

# Event 4: White Peace Offer (Received)
country_event = {
	id = plus_cs.4
	title = plus_cs.4.t
	desc = plus_cs.4.d
	picture = GFX_report_event_generic_sign_treaty2
	is_triggered_only = yes
	
	trigger = {
		is_puppet = no
		NOT = { tag = FROM } # SELF-PEACE GUARD
		# CIVIL WAR GUARD: Блокирует ванильные гражданки (Кайзер и т.д.)
		# Пропускает: обычные страны + модовые гражданки
		OR = {
			has_civil_war = no
			has_country_flag = plus_cs_had_civil_war
		}
	}

	immediate = {
		plus_cs_calc_white_peace_chance = yes
		log = "[PEACE_LOG] EVENT: [ROOT.GetName] considering WHITE PEACE from [FROM.GetName]. Score: [?plus_cs_final_calc_score]."
	}

	# Option A: Accept White Peace
	option = {
		name = plus_cs.4.a
		ai_chance = {
			base = 3 # Micro-chance — never zero acceptance
			
			# === SCORE-BASED PROBABILITY ===
			modifier = { add = 8 check_variable = { plus_cs_final_calc_score > 0 } }
			modifier = { add = 12 check_variable = { plus_cs_final_calc_score > 20 } }
			modifier = { add = 15 check_variable = { plus_cs_final_calc_score > 40 } }
			modifier = { add = 20 check_variable = { plus_cs_final_calc_score > 60 } }
			modifier = { add = 20 check_variable = { plus_cs_final_calc_score > 80 } }

			# --- NEGATIVE SCORE PENALTIES ---
			modifier = { factor = 0.5 check_variable = { plus_cs_final_calc_score < -20 } }
			modifier = { factor = 0.2 check_variable = { plus_cs_final_calc_score < -50 } }
			modifier = { factor = 0.05 check_variable = { plus_cs_final_calc_score < -100 } }
			modifier = { factor = 0.01 check_variable = { plus_cs_final_calc_score < -200 } }

			# Winners: white peace = losing progress. Micro-chance to accept
			modifier = {
				factor = 0.08
				surrender_progress < 0.1
				NOT = { has_idea = internal_crisis_low }
				NOT = { has_idea = internal_crisis_medium }
				NOT = { has_idea = internal_crisis_high }
				NOT = { has_dynamic_modifier = homefront_collapse }
				NOT = { has_dynamic_modifier = total_exhaustion }
			}
			
			# Losers: white peace is a gift — very likely to accept
			modifier = { add = 30 surrender_progress > 0.2 }
			modifier = { add = 40 surrender_progress > 0.4 }
			modifier = { add = 25 surrender_progress > 0.6 }
			
			# Winner in crisis: white peace becomes attractive
			modifier = {
				add = 20
				OR = {
					has_idea = internal_crisis_high
					has_dynamic_modifier = homefront_collapse
					has_dynamic_modifier = total_exhaustion
				}
			}
			
			# Negative score: less likely but NEVER zero
			modifier = { factor = 0.5 check_variable = { plus_cs_final_calc_score < -20 } }
			modifier = { factor = 0.4 check_variable = { plus_cs_final_calc_score < -50 } }
			modifier = { factor = 0.2 check_variable = { plus_cs_final_calc_score < -80 } }
		}

		# Execute White Peace
		# Execute White Peace
		log = "[PEACE_LOG] OUTCOME: [ROOT.GetName] ACCEPTED White Peace from [FROM.GetName]."
		
		# ALLIANCE PEACE FIX
		# If ROOT is Faction Leader, we peace out OUR ENTIRE FACTION against FROM
		if = {
			limit = { is_faction_leader = yes }
			white_peace = FROM
			set_country_flag = { flag = plus_cs_signed_separate_peace days = 360 }
			add_timed_idea = { idea = plus_cs_separate_peace_blocker days = 365 }
			every_other_country = {
				limit = { is_in_faction_with = ROOT }
				white_peace = FROM
				set_country_flag = { flag = plus_cs_signed_separate_peace days = 360 }
				add_timed_idea = { idea = plus_cs_separate_peace_blocker days = 365 }
			}
			# Centralized truce for entire faction (handles symmetric + puppets)
			plus_cs_set_faction_truce = yes
		}
		else = {
			# НОВОЕ: Сохраняем контроль третьих сторон перед белым миром
			plus_cs_save_third_party_control = yes
			
			white_peace = FROM
			leave_faction = yes
			
			set_country_flag = { flag = plus_cs_signed_separate_peace days = 360 }
			add_timed_idea = { idea = plus_cs_separate_peace_blocker days = 365 }
			# Centralized mutual truce (handles symmetric + puppets)
			plus_cs_set_mutual_truce = yes
			
			# НОВОЕ: Восстанавливаем контроль третьих сторон
			plus_cs_restore_third_party_control = yes
		}
		
		FROM = {
			clr_country_flag = plus_cs_waiting_response
			# Notify sender (NEWS EVENT FIX)
			news_event = { id = plus_cs.5 }
			add_timed_idea = { idea = plus_cs_separate_peace_blocker days = 365 }
		}
		
		add_political_power = 25
		add_stability = 0.05
	}

	# Option B: Refuse
	option = {
		name = plus_cs.4.b
		ai_chance = {
			base = 40
			
			# === WINNER: refuse white peace aggressively ===
			modifier = {
				add = 400
				FROM = { surrender_progress > 0.1 }
				surrender_progress < 0.1
			}
			
			# === LOSER: unlikely to refuse, but micro-chance preserved ===
			modifier = {
				factor = 0.1
				surrender_progress > 0.25
			}
			modifier = {
				factor = 0.15
				surrender_progress > 0.4
			}
			
			# Ideology: totalitarians refuse white peace easier
			modifier = {
				factor = 1.8
				OR = { has_government = fascism has_government = communism }
			}
			
			# Democrats: pragmatic, accept white peace more
			modifier = {
				factor = 0.5
				has_government = democratic
				has_country_flag = { flag = plus_cs_at_war_start days > 365 }
			}
			
			# We have territorial claims → refuse
			modifier = {
				add = 40
				FROM = { any_owned_state = { is_claimed_by = ROOT } }
			}
			
			# We have cores under enemy control → refuse
			modifier = {
				add = 60
				FROM = { any_owned_state = { is_core_of = ROOT } }
			}
			
			# Exhaustion → very reluctant to refuse but micro-chance
			modifier = {
				factor = 0.05
				OR = {
					has_country_flag = plus_cs_exhausted
					has_country_flag = plus_cs_urgent_peace
				}
			}
			
			# Stalemate: reluctant to refuse but micro-chance
			modifier = {
				factor = 0.15
				has_country_flag = plus_cs_stalemate
			}
			
			# === WINNER IN CRISIS: less refuse ===
			modifier = {
				factor = 0.25
				OR = {
					has_idea = internal_crisis_high
					has_dynamic_modifier = homefront_collapse
					has_dynamic_modifier = total_exhaustion
				}
			}
			
			# Long war: refusing white peace is unpopular
			modifier = {
				factor = 0.5
				has_country_flag = { flag = plus_cs_at_war_start days > 730 }
			}
			modifier = {
				factor = 0.3
				has_country_flag = { flag = plus_cs_at_war_start days > 1095 }
			}
		}
		
		set_temp_variable = { penalty_stab = 0 }
		set_temp_variable = { penalty_ws = 0 }
		set_temp_variable = { penalty_pp = -10 }
		plus_cs_apply_scaled_penalty_from_vars = yes
		
		# White peace refusal → penalty during stalemate
		if = {
			limit = { has_country_flag = plus_cs_stalemate }
			add_stability = -0.05
			add_war_support = -0.05
			custom_effect_tooltip = plus_cs.4.b.tooltip_1
		}
		
		log = "[PEACE_LOG] OUTCOME: [ROOT.GetName] REFUSED White Peace from [FROM.GetName]."

		FROM = {
			clr_country_flag = plus_cs_waiting_response
			set_country_flag = { flag = plus_cs_peace_cooldown days = 30 }
			# Escalation tracking
			set_country_flag = plus_cs_offered_peace_before
			# BUGFIX: Notify sender about rejection
			news_event = { id = plus_cs.6 }
		}
	}
}
