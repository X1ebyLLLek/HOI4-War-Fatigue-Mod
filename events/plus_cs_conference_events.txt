add_namespace = plus_cs



# ==========================================

# Event 2: Political Preferences / Imperial Conference
country_event = {
	id = plus_cs.2
	title = plus_cs.2.t
	desc = plus_cs.2.d
	picture = GFX_report_event_imperial_conference
	is_triggered_only = yes
	
	trigger = {
		is_puppet = no
		NOT = { tag = FROM } # SELF-PEACE GUARD
	}

	immediate = {
		set_temp_variable = { cnt_og_vp = 0 }
		plus_cs_calculate_total_strategic_value = yes
		# Calculate acceptance chance
		plus_cs_calc_conditional_surrender_chance = yes
		log = "[PEACE_LOG] EVENT: [ROOT.GetName] considering CONDITIONAL SURRENDER from [FROM.GetName]. Score: [?plus_cs_final_calc_score]."
	}

	# Option A: Discuss Terms (Accept)
	option = {
		name = plus_cs.2.a
		ai_chance = {
			base = 5 # Chaos Factor
			
			# === WEIGHTED PROBABILITY ===
			# Score > 0: Increasing chance
			modifier = { add = 10 check_variable = { plus_cs_final_calc_score > 0 } }
			modifier = { add = 15 check_variable = { plus_cs_final_calc_score > 20 } }
			modifier = { add = 20 check_variable = { plus_cs_final_calc_score > 40 } }
			modifier = { add = 25 check_variable = { plus_cs_final_calc_score > 60 } }
			
			# Score < 0: Decreasing chance
			modifier = { factor = 0.5 check_variable = { plus_cs_final_calc_score < -20 } }
			modifier = { factor = 0.5 check_variable = { plus_cs_final_calc_score < -50 } } # Relaxed from 0.1
			modifier = { factor = 0.1 check_variable = { plus_cs_final_calc_score < -80 } } # Relaxed from 0.0
		}

		if = {
			limit = {
				is_faction_leader = no
			}
			log = "[PEACE_LOG] OUTCOME: [ROOT.GetName] ACCEPTED Conditional Surrender from [FROM.GetName] (Minor Flow)."
			ROOT = {
				save_event_target_as = current_winner
				plus_cs_apply_peace_decisions_to_losers = yes
				every_subject_country = {
					save_event_target_as = current_winner
					plus_cs_apply_peace_decisions_to_losers = yes
				}
			}
		}
		else = {
			log = "[PEACE_LOG] OUTCOME: [ROOT.GetName] ACCEPTED Conditional Surrender from [FROM.GetName] (Major Flow)."
			ROOT = {
				save_event_target_as = current_winner
				plus_cs_apply_conditional_demands = yes
				
				every_other_country = {
					limit = {
						is_in_faction_with = ROOT
						has_war_with = FROM
					}
					save_event_target_as = current_winner
					plus_cs_apply_conditional_demands = yes
				}
			}
		}

		if = {
			limit = {
				is_faction_leader = no
			}
			plus_cs_save_third_party_control = yes
			
			white_peace = FROM
			set_country_flag = { flag = plus_cs_signed_separate_peace days = 360 }
			# Centralized extended truce (720 days, handles symmetric + puppets)
			plus_cs_set_extended_truce = yes
			
			plus_cs_restore_third_party_control = yes
		}
		else = {
			if = { limit = { has_capitulated = yes } remove_from_faction = yes }
			every_other_country = {
				limit = {
					is_in_faction_with = ROOT
					has_war_with = FROM
					has_capitulated = yes
				}
				remove_from_faction = yes
			}
			plus_cs_save_third_party_control = yes
			
			if = { 
				limit = { has_capitulated = no }
				white_peace = FROM
				set_country_flag = { flag = plus_cs_signed_separate_peace days = 360 }
			}
			every_other_country = {
				limit = {
					is_in_faction_with = ROOT
					has_war_with = FROM
					has_capitulated = no
				}
				white_peace = FROM
				set_country_flag = { flag = plus_cs_signed_separate_peace days = 360 }
			}
			# Centralized extended truce for faction (720 days, handles symmetric + puppets)
			plus_cs_set_extended_truce = yes
			
			if = {
				limit = {
					FROM = { is_faction_leader = no }
				}
				FROM = {
					leave_faction = yes
					add_timed_idea = {
						idea = conditional_surrender
						days = 720
					}
				}
			}
			FROM = {
				clr_country_flag = plus_cs_waiting_response
			}
		}
	}

	# Option B: Refuse
	option = {
		name = plus_cs.2.b
		ai_chance = { 
			base = 50
			
			# Win Harder: If Winning, definitely refuse
			modifier = {
				add = 50
				surrender_progress < 0.1
				FROM = { surrender_progress > 0.1 }
			}
			
			# Ideology: totalitarians want total victory
			modifier = {
				factor = 2.0
				OR = { has_government = fascism has_government = communism }
			}
			
			# Democrats: more pragmatic about accepting terms
			modifier = {
				factor = 0.5
				has_government = democratic
				has_country_flag = { flag = plus_cs_at_war_start days > 730 }
			}
			
			# We have claims/cores on the sender → refuse (we want more)
			modifier = {
				add = 40
				FROM = { any_owned_state = { OR = { is_claimed_by = ROOT is_core_of = ROOT } } }
			}
			
			# Enemy has capitulated → why refuse their surrender??
			modifier = {
				factor = 0.1
				FROM = { has_capitulated = yes }
			}
			
			# Enemy lost capital → they're broken, accept
			modifier = {
				factor = 0.5
				FROM = { capital_scope = { NOT = { is_controlled_by = FROM } } }
			}
			
			# Our own exhaustion: we want this war over
			modifier = {
				factor = 0.2
				OR = {
					has_country_flag = plus_cs_exhausted
					has_country_flag = plus_cs_stalemate
				}
			}
			
			# Long war fatigue
			modifier = {
				factor = 0.5
				has_country_flag = { flag = plus_cs_at_war_start days > 1095 }
			}
		}
		
		
		# F2 FIX: Было +stab/+ws (награда за отказ!). Теперь политическая цена.
		set_temp_variable = { penalty_stab = -0.02 }
		set_temp_variable = { penalty_ws = -0.05 }
		set_temp_variable = { penalty_pp = 0 }
		plus_cs_apply_scaled_penalty_from_vars = yes
		
		log = "[PEACE_LOG] OUTCOME: [ROOT.GetName] REFUSED Conditional Surrender from [FROM.GetName]."

		if = {
			limit = {
				OR = {
					has_country_flag = plus_cs_war_strain
					has_country_flag = plus_cs_stalemate
				}
			}
			set_temp_variable = { penalty_stab = -0.10 }
			set_temp_variable = { penalty_ws = -0.10 }
			set_temp_variable = { penalty_pp = 0 }
			plus_cs_apply_scaled_penalty_from_vars = yes
			custom_effect_tooltip = plus_cs.2.b.tooltip_1
		}
		if = {
			limit = {
				OR = {
					has_country_flag = plus_cs_exhausted
					has_country_flag = plus_cs_urgent_peace
				}
			}
			set_temp_variable = { penalty_stab = -0.15 }
			set_temp_variable = { penalty_ws = -0.15 }
			set_temp_variable = { penalty_pp = 0 }
			plus_cs_apply_scaled_penalty_from_vars = yes
			custom_effect_tooltip = plus_cs.2.b.tooltip_2
		}
		
		FROM = {
			clr_country_flag = plus_cs_waiting_response
			set_country_flag = { flag = plus_cs_peace_cooldown days = 45 }
			set_country_flag = plus_cs_offered_peace_before
			news_event = { id = plus_cs.6 }
		}
	}
}
