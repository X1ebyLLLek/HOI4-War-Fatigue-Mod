add_namespace = plus_cs



# ==========================================
# PEACE NEGOTIATION EVENTS (OFFER)
# ==========================================

# Event 1: Negotiation Offer (Received by Target)
country_event = {
	id = plus_cs.1
	title = plus_cs.1.t
	desc = plus_cs.1.d
	picture = GFX_report_event_generic_conference
	is_triggered_only = yes
	
	immediate = {
		plus_cs_calc_negotiation_chance = yes
		# Calculate reparation bonus from SENDER's crisis state
		FROM = { plus_cs_calc_reparation_bonus = yes }
	}

	trigger = {
		is_puppet = no
		NOT = { tag = FROM } # SELF-PEACE GUARD: страна не может вести переговоры сама с собой
	}

	# Option A: Accept Negotiation
	option = {
		name = plus_cs.1.a
		ai_chance = {
			base = 5 # Always a small chance (Chaos Factor)
			
			# === WEIGHTED PROBABILITY ===
			# Score > 0: Increasing chance
			modifier = { add = 10 check_variable = { plus_cs_final_calc_score > 0 } }
			modifier = { add = 15 check_variable = { plus_cs_final_calc_score > 20 } }
			modifier = { add = 20 check_variable = { plus_cs_final_calc_score > 40 } }
			modifier = { add = 25 check_variable = { plus_cs_final_calc_score > 60 } }
			modifier = { add = 25 check_variable = { plus_cs_final_calc_score > 80 } }
			
			# Score < 0: Decreasing chance (but not impossible unless extreme)
			modifier = { factor = 0.5 check_variable = { plus_cs_final_calc_score < -20 } }
			modifier = { factor = 0.1 check_variable = { plus_cs_final_calc_score < -50 } }
			modifier = { factor = 0.01 check_variable = { plus_cs_final_calc_score < -80 } }
			
			# === REPARATION BONUS ===
			# If sender is in crisis, they offer bonus = more likely to accept
			# Using FACTOR to multiply existing chance, not absolute add
			modifier = { factor = 1.1 check_variable = { FROM.plus_cs_reparation_bonus > 0 } }   # +10%
			modifier = { factor = 1.2 check_variable = { FROM.plus_cs_reparation_bonus > 20 } }  # +20% (total 1.32x)
			modifier = { factor = 1.3 check_variable = { FROM.plus_cs_reparation_bonus > 40 } }  # +30% (total 1.72x)
			modifier = { factor = 1.4 check_variable = { FROM.plus_cs_reparation_bonus > 80 } }  # +40% (total 2.40x)
		}

		# === REPARATION BONUS REWARD ===
		# Recipient gets PP based on sender's crisis level
		if = {
			limit = { check_variable = { FROM.plus_cs_reparation_bonus > 0 } }
			add_political_power = FROM.plus_cs_reparation_bonus
			custom_effect_tooltip = plus_cs_reparation_bonus_tt
		}

		# Calculate Strategic Value of owned states
		plus_cs_calculate_total_strategic_value = yes

		if = {
			limit = { FROM = { is_faction_leader = no } }
			# WINNER LOOP: FROM and its puppets
			FROM = {
				save_event_target_as = current_winner
				plus_cs_apply_peace_decisions_to_losers = yes
				every_subject_country = {
					save_event_target_as = current_winner
					plus_cs_apply_peace_decisions_to_losers = yes
				}
			}
		}
		else = {
			# WINNER LOOP: FROM's faction
			FROM = {
				save_event_target_as = current_winner
				plus_cs_apply_peace_decisions_to_losers = yes
				
				every_other_country = {
					limit = { is_in_faction_with = FROM has_war_with = ROOT }
					save_event_target_as = current_winner
					plus_cs_apply_peace_decisions_to_losers = yes
				}
			}
		}
		
		# Separate Peace handling
		if = {
			limit = {
				is_faction_leader = no
			}
			# НОВОЕ: Сохраняем контроль третьих сторон перед белым миром
			plus_cs_save_third_party_control = yes
			
			white_peace = FROM
			set_country_flag = { flag = plus_cs_signed_separate_peace days = 360 }
			add_timed_idea = { idea = plus_cs_separate_peace_blocker days = 365 }
			# Centralized mutual truce (handles symmetric + puppets)
			plus_cs_set_mutual_truce = yes
			
			# НОВОЕ: Восстанавливаем контроль третьих сторон после белого мира
			plus_cs_restore_third_party_control = yes
		}
		else = {
			every_enemy_country = {
				limit = {
					is_in_faction_with = FROM
					has_capitulated = yes
				}
				remove_from_faction = yes
			}
			# НОВОЕ: Сохраняем контроль третьих сторон
			plus_cs_save_third_party_control = yes
			
			every_enemy_country = {
				limit = {
					is_in_faction_with = FROM
					has_capitulated = no
				}
				white_peace = ROOT
			}
			# Extended truce for faction peace (720 days, handles symmetric + puppets)
			plus_cs_set_extended_truce = yes
			
			# НОВОЕ: Восстанавливаем контроль третьих сторон
			plus_cs_restore_third_party_control = yes
		}

		if = {
			limit = {
				is_faction_leader = no
			}
			leave_faction = yes
			add_timed_idea = {
				idea = conditional_surrender
				days = 720
			}
		}
		FROM = {
			clr_country_flag = plus_cs_waiting_response
		}
	}

	# Option B: Decline Negotiation
	option = {
		name = plus_cs.1.b
		ai_chance = {
			base = 50
			
			# Win Harder: If Winning, almost definitely Refuse
			modifier = {
				add = 100
				surrender_progress < 0.1
				FROM = { surrender_progress > 0.1 }
			}
			
			# Ideology: totalitarians demand total victory
			modifier = {
				factor = 2.0
				OR = {
					has_government = fascism
					has_government = communism
				}
			}
			
			# Democrats: pragmatic after long wars
			modifier = {
				factor = 0.5
				has_government = democratic
				has_country_flag = { flag = plus_cs_at_war_start days > 730 }
			}
			
			# Territory claims: we want their land
			modifier = {
				add = 60
				FROM = {
					any_owned_state = { is_claimed_by = ROOT }
				}
			}
			
			# We have cores under enemy control
			modifier = {
				add = 80
				FROM = { any_owned_state = { is_core_of = ROOT } }
			}
			
			# Enemy has capitulated: absurd to refuse their surrender
			modifier = {
				factor = 0.05
				FROM = { has_capitulated = yes }
			}
			
			# Enemy lost capital: they're broken
			modifier = {
				factor = 0.3
				FROM = { capital_scope = { NOT = { is_controlled_by = FROM } } }
			}
			
			# Exhaustion: accept any peace
			modifier = {
				factor = 0
				OR = {
					has_country_flag = plus_cs_exhausted
					has_country_flag = plus_cs_urgent_peace
					has_country_flag = plus_cs_stalemate
				}
			}
			
			# Long war: war weariness
			modifier = {
				factor = 0.5
				has_country_flag = { flag = plus_cs_at_war_start days > 1095 }
			}
		}
		
		add_war_support = 0.05
		add_stability = 0.02
		
		if = {
			limit = {
				OR = {
					has_country_flag = plus_cs_war_strain
					has_country_flag = plus_cs_stalemate
				}
			}
			add_war_support = -0.10
			add_stability = -0.05
			custom_effect_tooltip = plus_cs.1.b.tooltip_1
		}
		if = {
			limit = {
				OR = {
					has_country_flag = plus_cs_exhausted
					has_country_flag = plus_cs_urgent_peace
				}
			}
			add_war_support = -0.15
			add_stability = -0.10
			custom_effect_tooltip = plus_cs.1.b.tooltip_2
		}
		
		FROM = {
			clr_country_flag = plus_cs_waiting_response
			set_country_flag = { flag = plus_cs_peace_cooldown days = 45 }
			news_event = { id = plus_cs.6 }
		}
		
		# === ESCALATION SYSTEM ===
		# Repeated rejections have escalating political costs
		if = {
			limit = { FROM = { has_country_flag = plus_cs_offered_peace_before } }
			# Second+ rejection: public frustration (war weariness)
			add_stability = -0.05
			add_war_support = -0.03
			FROM = { set_country_flag = plus_cs_offered_peace_twice }
		}
		FROM = { set_country_flag = plus_cs_offered_peace_before }
	}
}
