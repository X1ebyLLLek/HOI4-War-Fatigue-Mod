add_namespace = plus_cs

# ==========================================
# PEACE NEGOTIATION EVENTS (OFFER)
# ==========================================

# Event 1: Negotiation Offer (Received by Target)
country_event = {
	id = plus_cs.1
	title = plus_cs.1.t
	desc = plus_cs.1.d
	picture = GFX_report_event_generic_conference
	is_triggered_only = yes
	
	immediate = {
		plus_cs_calc_negotiation_chance = yes
	}

	trigger = { is_puppet = no }

	# Option A: Accept Negotiation
	option = {
		name = plus_cs.1.a
		ai_chance = {
			base = 5 # Always a small chance (Chaos Factor)
			
			# === WEIGHTED PROBABILITY ===
			# Score > 0: Increasing chance
			modifier = { add = 10 check_variable = { plus_cs_final_calc_score > 0 } }
			modifier = { add = 15 check_variable = { plus_cs_final_calc_score > 20 } }
			modifier = { add = 20 check_variable = { plus_cs_final_calc_score > 40 } }
			modifier = { add = 25 check_variable = { plus_cs_final_calc_score > 60 } }
			modifier = { add = 25 check_variable = { plus_cs_final_calc_score > 80 } }
			
			# Score < 0: Decreasing chance (but not impossible unless extreme)
			modifier = { factor = 0.5 check_variable = { plus_cs_final_calc_score < -20 } }
			modifier = { factor = 0.1 check_variable = { plus_cs_final_calc_score < -50 } }
			modifier = { factor = 0.01 check_variable = { plus_cs_final_calc_score < -80 } }
		}

		# Calculate Strategic Value of owned states
		plus_cs_calculate_total_strategic_value = yes

		if = {
			limit = {
				FROM = { is_faction_leader = no }
			}
			every_country = {
				limit = {
					OR = {
						AND = {
							ROOT = { is_faction_leader = no }
							OR = {
								tag = ROOT
								is_puppet_of = ROOT
							}
						}
						AND = {
							ROOT = { is_faction_leader = yes }
							is_in_faction_with = ROOT
						}
					}
				}
				every_owned_state = {
					every_country = {
						limit = {
							OR = {
								tag = FROM
								is_puppet_of = FROM
							}
						}
						set_variable = { og_vp@PREV.owner = cnt_og_vp@PREV.owner }
						activate_targeted_decision = { target = PREV.owner decision = peace_war_reparations }
						activate_targeted_decision = { target = PREV.owner decision = peace_disarm }
						activate_targeted_decision = { target = PREV decision = peace_take_state }
						activate_targeted_decision = { target = PREV decision = peace_puppet }
						activate_targeted_decision = { target = PREV decision = peace_set_dmz }
						activate_targeted_decision = { target = PREV decision = peace_liberate }
						activate_targeted_decision = { target = PREV decision = peace_claim }
					}
					set_variable = { state_taken = 0 }
				}
			}
		}
		# OPTIMIZATION: Removed O(N^3) Loop.
		# Instead of looping every_country > every_owned_state > every_country,
		# We check only relevant neighbors/enemies.
		else = {
			every_country = {
				# Limit 1: Only consider relevant countries (Majors, Neighbors, Enemies)
				limit = {
					OR = {
						AND = {
							ROOT = { is_faction_leader = no }
							OR = {
								tag = ROOT
								is_puppet_of = ROOT
							}
						}
						AND = {
							ROOT = { is_faction_leader = yes }
							is_in_faction_with = ROOT
						}
					}
				}
				every_owned_state = {
					# Limit 2: Do not iterate ALL countries for every state. 
					# Use scoped checks.
					every_country = {
						limit = {
							is_in_faction_with = FROM
							has_war_with = ROOT
							# OPTIMIZATION: Only if they actually have a claim or core here
							PREV = {
								OR = {
									is_core_of = PREV.PREV
									is_claimed_by = PREV.PREV
								}
							}
						}
						
						set_variable = { og_vp@PREV.PREV = cnt_og_vp@PREV.PREV }
						activate_targeted_decision = { target = PREV.PREV decision = peace_war_reparations }
						activate_targeted_decision = { target = PREV.PREV decision = peace_disarm }
						activate_targeted_decision = { target = PREV decision = peace_take_state }
						activate_targeted_decision = { target = PREV decision = peace_set_dmz }
						activate_targeted_decision = { target = PREV decision = peace_puppet }
						activate_targeted_decision = { target = PREV decision = peace_liberate }
						activate_targeted_decision = { target = PREV decision = peace_claim }
					}
					set_variable = { state_taken = 0 }
				}
			}
		}
		
		# Separate Peace handling
		if = {
			limit = {
				is_faction_leader = no
			}
			# НОВОЕ: Сохраняем контроль третьих сторон перед белым миром
			plus_cs_save_third_party_control = yes
			
			white_peace = FROM
			set_country_flag = { flag = plus_cs_signed_separate_peace days = 360 }
			set_truce = {
				target = FROM
				days = 720
			}
			
			# НОВОЕ: Восстанавливаем контроль третьих сторон после белого мира
			plus_cs_restore_third_party_control = yes
		}
		else = {
			every_enemy_country = {
				limit = {
					is_in_faction_with = FROM
					has_capitulated = yes
				}
				remove_from_faction = yes
			}
			# НОВОЕ: Сохраняем контроль третьих сторон
			plus_cs_save_third_party_control = yes
			
			every_enemy_country = {
				limit = {
					is_in_faction_with = FROM
					has_capitulated = no
				}
				white_peace = ROOT
				set_truce = {
					target = ROOT
					days = 720
				}
			}
			
			# НОВОЕ: Восстанавливаем контроль третьих сторон
			plus_cs_restore_third_party_control = yes
		}

		if = {
			limit = {
				is_faction_leader = no
			}
			leave_faction = yes
			add_timed_idea = {
				idea = conditional_surrender
				days = 720
			}
		}
		FROM = {
			clr_country_flag = plus_cs_waiting_response
		}
	}

	# Option B: Decline Negotiation
	option = {
		name = plus_cs.1.b
		ai_chance = {
			base = 50
			
			# Win Harder: If Winning, almost definitely Refuse
			modifier = {
				add = 100
				surrender_progress < 0.1
				FROM = { surrender_progress > 0.1 }
			}
			
			modifier = {
				factor = 5.0
				OR = {
					has_government = fascism
					has_government = communism
				}
			}
			modifier = {
				add = 100
				FROM = {
					any_owned_state = { is_claimed_by = ROOT }
				}
			}
			modifier = {
				factor = 0
				OR = {
					has_country_flag = plus_cs_exhausted
					has_country_flag = plus_cs_urgent_peace
					has_country_flag = plus_cs_stalemate
				}
			}
		}
		
		add_war_support = 0.05
		add_stability = 0.02
		
		if = {
			limit = {
				OR = {
					has_country_flag = plus_cs_war_strain
					has_country_flag = plus_cs_stalemate
				}
			}
			add_war_support = -0.10
			add_stability = -0.05
			custom_effect_tooltip = plus_cs.1.b.tooltip_1
		}
		if = {
			limit = {
				OR = {
					has_country_flag = plus_cs_exhausted
					has_country_flag = plus_cs_urgent_peace
				}
			}
			add_war_support = -0.15
			add_stability = -0.10
			custom_effect_tooltip = plus_cs.1.b.tooltip_2
		}
		
		FROM = {
			clr_country_flag = plus_cs_waiting_response
			news_event = { id = plus_cs.6 }
		}
	}
}
