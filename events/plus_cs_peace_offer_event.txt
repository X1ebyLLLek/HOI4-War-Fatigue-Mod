add_namespace = plus_cs



# ==========================================
# PEACE NEGOTIATION EVENTS (OFFER)
# ==========================================

# Event 1: Negotiation Offer (Received by Target)
country_event = {
	id = plus_cs.1
	title = plus_cs.1.t
	desc = plus_cs.1.d
	picture = GFX_report_event_generic_conference
	is_triggered_only = yes
	
	immediate = {
		plus_cs_calc_negotiation_chance = yes
		# Calculate reparation bonus from SENDER's crisis state
		FROM = { plus_cs_calc_reparation_bonus = yes }
	}

	trigger = {
		is_puppet = no
		NOT = { tag = FROM } # SELF-PEACE GUARD: страна не может вести переговоры сама с собой
	}

	# Option A: Accept Negotiation
	option = {
		name = plus_cs.1.a
		ai_chance = {
			base = 3 # Micro-chance (Chaos Factor) — never zero
			
			# === SCORE-BASED PROBABILITY ===
			modifier = { add = 8 check_variable = { plus_cs_final_calc_score > 0 } }
			modifier = { add = 12 check_variable = { plus_cs_final_calc_score > 20 } }
			modifier = { add = 15 check_variable = { plus_cs_final_calc_score > 40 } }
			modifier = { add = 20 check_variable = { plus_cs_final_calc_score > 60 } }
			modifier = { add = 20 check_variable = { plus_cs_final_calc_score > 80 } }
			
			# Negative score: low but NEVER zero
			modifier = { factor = 0.5 check_variable = { plus_cs_final_calc_score < -20 } }
			modifier = { factor = 0.15 check_variable = { plus_cs_final_calc_score < -50 } }
			modifier = { factor = 0.05 check_variable = { plus_cs_final_calc_score < -100 } }
			modifier = { factor = 0.01 check_variable = { plus_cs_final_calc_score < -200 } }
			
			# === WINNER GREED ===
			# If I'm winning, I want MORE than conditional terms — micro-chance to accept
			modifier = {
				factor = 0.15
				surrender_progress < 0.1
				FROM = { surrender_progress > 0.2 }
			}
			
			# === LOSER EAGERNESS ===
			# If I'm losing, conditional terms are salvation — very likely accept
			modifier = { add = 25 surrender_progress > 0.3 }
			modifier = { add = 40 surrender_progress > 0.5 }
			modifier = { add = 30 has_capitulated = yes }
			
			# === WINNER IN CRISIS ===
			# Winner who is struggling internally may accept lesser terms
			modifier = {
				add = 15
				surrender_progress < 0.1
				OR = {
					has_idea = internal_crisis_high
					has_dynamic_modifier = homefront_collapse
					has_dynamic_modifier = total_exhaustion
				}
			}
			
			# === REPARATION BONUS ===
			modifier = { factor = 1.1 check_variable = { FROM.plus_cs_reparation_bonus > 0 } }
			modifier = { factor = 1.15 check_variable = { FROM.plus_cs_reparation_bonus > 20 } }
			modifier = { factor = 1.2 check_variable = { FROM.plus_cs_reparation_bonus > 40 } }
			modifier = { factor = 1.3 check_variable = { FROM.plus_cs_reparation_bonus > 80 } }
		}

		# === REPARATION BONUS REWARD ===
		# Desperate country offers concessions: extra peace points for the winner
		if = {
			limit = { check_variable = { FROM.plus_cs_reparation_bonus > 0 } }
			add_to_variable = { vp@FROM = FROM.plus_cs_reparation_bonus }
			custom_effect_tooltip = plus_cs_reparation_bonus_tt
		}

		# Calculate Strategic Value of owned states
		plus_cs_calculate_total_strategic_value = yes

		# ============================================================
		# BUGFIX: SURRENDER INVERSION
		# When FROM has plus_cs_is_surrendering, FROM is the LOSER
		# and ROOT is the WINNER. Cannot use plus_cs_apply_peace_decisions_to_losers
		# because it uses ROOT to find losers, but here ROOT = winner.
		# Inline the decision activation directly.
		# ============================================================
		if = {
			limit = { FROM = { has_country_flag = plus_cs_is_surrendering } }
			
			# --- WINNER: ROOT (and allies) get decisions against FROM (loser) ---
			
			# Country-level decisions for ROOT against FROM
			ROOT = {
				save_event_target_as = current_winner
				set_variable = { og_vp@FROM = cnt_og_vp@FROM }
				activate_targeted_decision = { target = FROM decision = peace_war_reparations }
				activate_targeted_decision = { target = FROM decision = peace_disarm }
			}
			# State-level decisions: iterate FROM's owned states
			FROM = {
				every_owned_state = {
					set_variable = { is_peace_state_target = 1 }
					event_target:current_winner = {
						activate_targeted_decision = { target = PREV decision = peace_take_state }
						activate_targeted_decision = { target = PREV decision = peace_set_dmz }
						activate_targeted_decision = { target = PREV decision = peace_puppet }
						activate_targeted_decision = { target = PREV decision = peace_liberate }
						activate_targeted_decision = { target = PREV decision = peace_claim }
					}
				}
				set_variable = { state_taken = 0 }
			}
			
			# ROOT's allies also get decisions (if ROOT is faction leader)
			if = {
				limit = { is_faction_leader = yes }
				every_other_country = {
					limit = { is_in_faction_with = ROOT has_war_with = FROM }
					save_event_target_as = current_winner
					set_variable = { og_vp@FROM = cnt_og_vp@FROM }
					activate_targeted_decision = { target = FROM decision = peace_war_reparations }
					activate_targeted_decision = { target = FROM decision = peace_disarm }
					FROM = {
						every_owned_state = {
							set_variable = { is_peace_state_target = 1 }
							event_target:current_winner = {
								activate_targeted_decision = { target = PREV decision = peace_take_state }
								activate_targeted_decision = { target = PREV decision = peace_set_dmz }
								activate_targeted_decision = { target = PREV decision = peace_puppet }
								activate_targeted_decision = { target = PREV decision = peace_liberate }
								activate_targeted_decision = { target = PREV decision = peace_claim }
							}
						}
					}
				}
			}
			else = {
				# ROOT's puppets also get decisions
				every_subject_country = {
					save_event_target_as = current_winner
					set_variable = { og_vp@FROM = cnt_og_vp@FROM }
					activate_targeted_decision = { target = FROM decision = peace_war_reparations }
					activate_targeted_decision = { target = FROM decision = peace_disarm }
					FROM = {
						every_owned_state = {
							set_variable = { is_peace_state_target = 1 }
							event_target:current_winner = {
								activate_targeted_decision = { target = PREV decision = peace_take_state }
								activate_targeted_decision = { target = PREV decision = peace_set_dmz }
								activate_targeted_decision = { target = PREV decision = peace_puppet }
								activate_targeted_decision = { target = PREV decision = peace_liberate }
								activate_targeted_decision = { target = PREV decision = peace_claim }
							}
						}
					}
				}
			}
			
			# White peace — FROM (loser) makes peace with ROOT (winner)
			plus_cs_save_third_party_control = yes
			white_peace = FROM
			plus_cs_set_mutual_truce = yes
			plus_cs_restore_third_party_control = yes
			
			FROM = {
				clr_country_flag = plus_cs_is_surrendering
				clr_country_flag = plus_cs_waiting_response
				leave_faction = yes
				add_timed_idea = {
					idea = conditional_surrender
					days = 720
				}
				add_timed_idea = {
					idea = plus_cs_separate_peace_blocker
					days = 720
				}
			}
		}
		# ============================================================
		# STANDARD FLOW: FROM is the WINNER, ROOT is the LOSER
		# (Player proposed peace, bot accepted)
		# ============================================================
		else = {
		save_event_target_as = current_loser
		
		if = {
			limit = { FROM = { is_faction_leader = no } }
			# WINNER LOOP: FROM and its puppets
			FROM = {
				save_event_target_as = current_winner
				plus_cs_apply_peace_decisions_to_losers = yes
				every_subject_country = {
					save_event_target_as = current_winner
					plus_cs_apply_peace_decisions_to_losers = yes
				}
			}
		}
		else = {
			# WINNER LOOP: FROM's faction
			FROM = {
				save_event_target_as = current_winner
				plus_cs_apply_peace_decisions_to_losers = yes
				
				every_other_country = {
					limit = { is_in_faction_with = FROM has_war_with = ROOT }
					save_event_target_as = current_winner
					plus_cs_apply_peace_decisions_to_losers = yes
				}
			}
		}
		
		# Separate Peace handling
		if = {
			limit = {
				is_faction_leader = no
			}
			# НОВОЕ: Сохраняем контроль третьих сторон перед белым миром
			plus_cs_save_third_party_control = yes
			
			white_peace = FROM
			set_country_flag = { flag = plus_cs_signed_separate_peace days = 360 }
			add_timed_idea = { idea = plus_cs_separate_peace_blocker days = 365 }
			# Centralized mutual truce (handles symmetric + puppets)
			plus_cs_set_mutual_truce = yes
			
			# НОВОЕ: Восстанавливаем контроль третьих сторон после белого мира
			plus_cs_restore_third_party_control = yes
		}
		else = {
			# Peace out all participants robustly. 
			# ROOT is Loser Faction Leader, FROM is Winner.
			# We must ensure all active participants on both sides sign peace, 
			# including capitulated ones so they can interact with the mod GUI.
			
			# 1. Kick Capitulated LOSERS from the Loser faction (ROOT's allies)
			# This leaves them at war with the Winner, instantly triggering a 
			# vanilla peace conference on the next daily tick for total annexation!
			every_other_country = {
				limit = {
					is_in_faction_with = ROOT
					has_capitulated = yes
				}
				remove_from_faction = yes
			}
			
			# НОВОЕ: Сохраняем контроль третьих сторон
			plus_cs_save_third_party_control = yes
			
			# 2. Winner's faction makes peace with the Loser (ROOT)
			every_enemy_country = {
				limit = {
					is_in_faction_with = FROM
				}
				white_peace = ROOT
			}
			
			# 3. Loser's SURVIVING active faction members make peace with Winner's faction
			every_other_country = {
				limit = {
					is_in_faction_with = ROOT
					has_war_with = FROM
					has_capitulated = no # Ensure they haven't capitulated
				}
				white_peace = FROM # Peace with the Winner
				
				# Peace with the Winner's active faction
				every_enemy_country = {
					limit = {
						is_in_faction_with = FROM
					}
					white_peace = PREV
				}
			}
			
			# BUGFIX: Заключаем мир между самим FROM (Winner) и ROOT (Loser)
			white_peace = FROM
			
			# Extended truce for faction peace (720 days, handles symmetric + puppets)
			plus_cs_set_extended_truce = yes
			
			# НОВОЕ: Восстанавливаем контроль третьих сторон
			plus_cs_restore_third_party_control = yes
		}

		if = {
			limit = {
				is_faction_leader = no
			}
			leave_faction = yes
			add_timed_idea = {
				idea = conditional_surrender
				days = 720
			}
		}
		
		# BUGFIX: Ensure faction leaders also get peace blocker
		if = {
			limit = { is_faction_leader = yes }
			add_timed_idea = { idea = plus_cs_separate_peace_blocker days = 365 }
		}

		FROM = {
			clr_country_flag = plus_cs_waiting_response
		}
		
		} # end else (standard flow)
	}

	# Option B: Decline Negotiation
	option = {
		name = plus_cs.1.b
		ai_chance = {
			base = 40
			
			# === WINNER GREED: demand more! ===
			modifier = {
				add = 150
				surrender_progress < 0.1
				FROM = { surrender_progress > 0.1 }
			}
			# Crushing enemy — demand unconditional, not conditional
			modifier = {
				add = 100
				surrender_progress < 0.05
				FROM = { surrender_progress > 0.4 }
			}
			
			# Ideology: totalitarians demand total victory
			modifier = {
				factor = 1.8
				OR = {
					has_government = fascism
					has_government = communism
				}
			}
			
			# Democrats: pragmatic after long wars
			modifier = {
				factor = 0.5
				has_government = democratic
				has_country_flag = { flag = plus_cs_at_war_start days > 730 }
			}
			
			# Territory claims: we want their land
			modifier = {
				add = 60
				FROM = {
					any_owned_state = { is_claimed_by = ROOT }
				}
			}
			
			# We have cores under enemy control
			modifier = {
				add = 80
				FROM = { any_owned_state = { is_core_of = ROOT } }
			}
			
			# Enemy has capitulated: very unlikely to refuse their surrender (but micro-chance)
			modifier = {
				factor = 0.03
				FROM = { has_capitulated = yes }
			}
			
			# Enemy lost capital: they're broken
			modifier = {
				factor = 0.25
				FROM = { capital_scope = { NOT = { is_controlled_by = FROM } } }
			}
			
			# === LOSER: less likely to refuse (but micro-chance preserved) ===
			modifier = {
				factor = 0.3
				surrender_progress > 0.3
			}
			modifier = {
				factor = 0.2
				surrender_progress > 0.5
			}
			
			# Exhaustion: strongly prefer peace (but not factor 0!)
			modifier = {
				factor = 0.05
				OR = {
					has_country_flag = plus_cs_exhausted
					has_country_flag = plus_cs_urgent_peace
				}
			}
			
			# Stalemate: less refuse but not zero
			modifier = {
				factor = 0.2
				has_country_flag = plus_cs_stalemate
			}
			
			# === WINNER IN CRISIS: more willing to accept (less refuse) ===
			modifier = {
				factor = 0.3
				OR = {
					has_idea = internal_crisis_high
					has_dynamic_modifier = homefront_collapse
					has_dynamic_modifier = total_exhaustion
				}
			}
			
			# Long war: war weariness
			modifier = {
				factor = 0.5
				has_country_flag = { flag = plus_cs_at_war_start days > 1095 }
			}
		}
		
		# F2 FIX removed: No stability/war support penalty for declining peace with preferences.
		
		FROM = {
			clr_country_flag = plus_cs_waiting_response
			set_country_flag = { flag = plus_cs_peace_cooldown days = 45 }
			news_event = { id = plus_cs.6 }
		}
		
		# === ESCALATION SYSTEM ===
		# Escalating tracking (penalties removed per user request)
		if = {
			limit = { FROM = { has_country_flag = plus_cs_offered_peace_before } }
			FROM = { set_country_flag = plus_cs_offered_peace_twice }
		}
		FROM = { set_country_flag = plus_cs_offered_peace_before }
	}
}
