# ==========================================
# SUBMISSION EVENTS (201-205)
# ==========================================

country_event = {
	id = plus_cs.201
	title = plus_cs.submission.1.t
	desc = plus_cs.submission.1.d
	picture = GFX_report_event_generic_conference
	is_triggered_only = yes
	
	trigger = {
		# Gating: Only after Civil War
		has_country_flag = plus_cs_had_civil_war
		is_puppet = no
	}
	
	immediate = {
		plus_cs_calc_submission_chance = yes
		log = "[PEACE_LOG] EVENT: [ROOT.GetName] considering SUBMISSION from [FROM.GetName]. Score: [?plus_cs_final_calc_score]."
	}

	# Option A: Accept Submission (Puppet)
	option = {
		name = plus_cs.submission.1.a
		ai_chance = {
			base = 5 # Chaos Factor
			
			# === WEIGHTED PROBABILITY ===
			# Score > 0: Increasing chance
			modifier = { add = 10 check_variable = { plus_cs_final_calc_score > 0 } }
			modifier = { add = 15 check_variable = { plus_cs_final_calc_score > 20 } }
			modifier = { add = 20 check_variable = { plus_cs_final_calc_score > 40 } }
			modifier = { add = 25 check_variable = { plus_cs_final_calc_score > 60 } }
			modifier = { add = 25 check_variable = { plus_cs_final_calc_score > 80 } }
			
			# Score < 0: Decreasing chance
			modifier = { factor = 0.5 check_variable = { plus_cs_final_calc_score < -20 } }
			modifier = { factor = 0.1 check_variable = { plus_cs_final_calc_score < -50 } }
			modifier = { factor = 0.01 check_variable = { plus_cs_final_calc_score < -80 } }
		}
		
		log = "[PEACE_LOG] OUTCOME: [ROOT.GetName] ACCEPTED Submission from [FROM.GetName] (Puppet)."

		# 1. Clear Flag on Sender immediately
		FROM = {
			clr_country_flag = plus_cs_waiting_response
		}
		
		# 2. News Event (FIXED ID)
		news_event = { id = plus_cs.220 }
		
		# 3. ANNEX OCCUPIED STATES (Winner keeps what they took)
		FROM = {
			every_owned_state = {
				limit = {
					is_controlled_by = ROOT
					NOT = { is_core_of = ROOT }
				}
				ROOT = { transfer_state = PREV }
			}
		}
		
		# 4. Save Control (Custom Logic)
		plus_cs_save_third_party_control = yes
		
		# 5. Puppet (ends war automatically in HOI4)
		puppet = FROM
		
		if = {
			limit = { is_in_faction = yes }
			FROM = { leave_faction = yes }
		}
		
		# Transfer Subjects (Steal them)
		# FIX: Scope to FROM (the loser) to find their subjects
		FROM = {
			every_subject_country = {
				# limit = { is_subject_of = PREV }
				set_autonomy = {
					target = ROOT
					autonomy_state = autonomy_puppet
				}
			}
		}
		
		# 7. Restore Control & Finalize
		plus_cs_restore_third_party_control = yes
		
		# 8. Notify Sender
		FROM = {
			country_event = { id = plus_cs.202 }
		}
		
		add_political_power = 50
	}

	# Option B: Refuse and Fight On
	option = {
		name = plus_cs.submission.1.b
		ai_chance = {
			base = 5
			
			# Score < 0: High chance to refuse
			modifier = { add = 15 check_variable = { plus_cs_final_calc_score < -20 } }
			modifier = { add = 25 check_variable = { plus_cs_final_calc_score < -40 } }
			modifier = { add = 50 check_variable = { plus_cs_final_calc_score < -60 } }
			
			# Score > 0: Low chance to refuse
			modifier = { factor = 0.1 check_variable = { plus_cs_final_calc_score > 20 } }
			modifier = { factor = 0.1 check_variable = { plus_cs_final_calc_score > 60 } } # Never zero
		}

		add_war_support = 0.05
		add_political_power = -50
		
		log = "[PEACE_LOG] OUTCOME: [ROOT.GetName] REFUSED Submission from [FROM.GetName]."
		
		FROM = {
			clr_country_flag = plus_cs_waiting_response
			# FIX: Add cooldown to prevent submission spam
			set_country_flag = { flag = plus_cs_peace_cooldown days = 30 }
			country_event = { id = plus_cs.203 }
		}
	}
	
	# Option C: Cede Claims (Territorial Concession)
	option = {
		name = plus_cs.submission.1.c
		trigger = {
			FROM = {
				any_owned_state = {
					OR = {
						is_claimed_by = ROOT
						is_core_of = ROOT
					}
				}
			}
		}
		ai_chance = {
			base = 5
			# Logic: Middle ground. If Logic Score is "Meh" (-20 to +20).
			modifier = { add = 20 check_variable = { plus_cs_final_calc_score > -20 } }
			modifier = { add = 20 check_variable = { plus_cs_final_calc_score < 20 } }
		}
		
		# 1. Clear Flags
		log = "[PEACE_LOG] OUTCOME: [ROOT.GetName] ACCEPTED Submission from [FROM.GetName] (Cede Claims)."
		FROM = {
			clr_country_flag = plus_cs_waiting_response
		}
		
		# 2. Transfer Land
		FROM = {
			every_owned_state = {
				limit = {
					OR = {
						is_claimed_by = ROOT
						is_core_of = ROOT
					}
				}
				ROOT = { transfer_state = PREV }
			}
		}
		
		# 3. Save Control
		plus_cs_save_third_party_control = yes
		
		# 4. End War
		white_peace = FROM
		
		# 5. Restore Control
		plus_cs_restore_third_party_control = yes
		
		# 6. Notify Sender
		FROM = {
			country_event = { id = plus_cs.205 }
		}
		
		add_political_power = 10
	}
}

# ==========================================
# VICTOR ULTIMATUM EVENT (250)
# Loser receives demand to surrender
# FROM = Victor (demanding surrender)
# ROOT = Loser (receiving ultimatum)
# ==========================================
country_event = {
	id = plus_cs.250
	title = plus_cs.250.t
	desc = plus_cs.250.d
	picture = GFX_report_event_generic_military_parade
	is_triggered_only = yes
	
	trigger = {
		is_puppet = no
	}
	
	immediate = {
		log = "[PEACE_LOG] EVENT: [ROOT.GetName] received ULTIMATUM from [FROM.GetName]."
	}

	# Option A: Submit (Become puppet)
	option = {
		name = plus_cs.250.a
		ai_chance = {
			base = 10
			
			# Capitulated: almost always submit
			modifier = { add = 80 has_capitulated = yes }
			
			# Lost capital: strongly consider submission
			modifier = { add = 40 capital_scope = { NOT = { is_controlled_by = ROOT } } }
			
			# High surrender = weakened resolve
			modifier = { add = 20 surrender_progress > 0.4 }
			modifier = { add = 30 surrender_progress > 0.6 }
			
			# Exhausted nations just want it to end
			modifier = { add = 40 has_country_flag = plus_cs_exhausted }
			modifier = { add = 25 has_dynamic_modifier = total_exhaustion }
			
			# Power imbalance
			modifier = { add = 30 strength_ratio = { tag = FROM ratio < 0.1 } }
			
			# Defeatist leader
			modifier = { add = 25 has_country_leader_with_trait = plus_cs_spirit_defeatist }
			
			# Ideology compatibility
			modifier = {
				add = 15
				OR = {
					AND = { has_government = democratic FROM = { has_government = democratic } }
					AND = { has_government = neutrality FROM = { has_government = neutrality } }
				}
			}
			
			# All allies dead
			modifier = {
				add = 30
				is_in_faction = yes
				NOT = {
					any_other_country = {
						is_in_faction_with = ROOT
						NOT = { has_capitulated = yes }
					}
				}
			}
			
			# Democracy: more willing to submit to avoid bloodshed
			modifier = { add = 15 has_government = democratic }
			
			# Fanatic: almost never accepts
			modifier = { factor = 0.05 has_country_leader_with_trait = plus_cs_spirit_fanatic }
			
			# Fascist/Communist: stubborn resistance
			modifier = { 
				factor = 0.3 
				OR = { has_government = fascism has_government = communism }
				NOT = { has_capitulated = yes }
			}
			
			# Living major ally: hope remains
			modifier = {
				factor = 0.2
				is_in_faction = yes
				any_other_country = {
					is_in_faction_with = ROOT
					NOT = { has_capitulated = yes }
					is_major = yes
				}
			}
		}
		
		log = "[PEACE_LOG] OUTCOME: [ROOT.GetName] SUBMITTED to ultimatum from [FROM.GetName]."
		
		# Identical to submission accept mechanics
		FROM = {
			clr_country_flag = plus_cs_waiting_response
		}
		
		news_event = { id = plus_cs.220 }
		
		# Annex occupied states
		every_owned_state = {
			limit = {
				is_controlled_by = FROM
				NOT = { is_core_of = FROM }
			}
			FROM = { transfer_state = PREV }
		}
		
		plus_cs_save_third_party_control = yes
		FROM = { puppet = ROOT }
		
		if = {
			limit = { is_in_faction = yes }
			leave_faction = yes
		}
		
		plus_cs_restore_third_party_control = yes
		
		FROM = {
			add_political_power = 50
			country_event = { id = plus_cs.202 }
		}
	}

	# Option B: Resist ("Never!")
	option = {
		name = plus_cs.250.b
		ai_chance = {
			base = 40
			
			# Winning or stalemate: easy to refuse
			modifier = { add = 60 surrender_progress < 0.2 }
			
			# Fanatic
			modifier = { factor = 3.0 has_country_leader_with_trait = plus_cs_spirit_fanatic }
			
			# Fascist/Communist resolve
			modifier = { factor = 2.0 OR = { has_government = fascism has_government = communism } }
			
			# Capitulated: hard to resist
			modifier = { factor = 0.1 has_capitulated = yes }
			
			# Losing badly: stubbornness is futile
			modifier = { factor = 0.3 surrender_progress > 0.5 }
		}
		
		log = "[PEACE_LOG] OUTCOME: [ROOT.GetName] RESISTED ultimatum from [FROM.GetName]."
		
		# Rally effect: rejected ultimatum inspires resistance
		add_war_support = 0.15
		add_stability = 0.05
		add_manpower = 25000
		
		# But there is international cost
		add_political_power = -25
		
		FROM = {
			clr_country_flag = plus_cs_waiting_response
			set_country_flag = { flag = plus_cs_peace_cooldown days = 60 }
		}
	}

	# Option C: Counter-propose (Conditional Surrender — keep independence)
	option = {
		name = plus_cs.250.c
		
		trigger = {
			NOT = { has_capitulated = yes }
			surrender_progress > 0.2
		}
		
		ai_chance = {
			base = 20
			
			# More likely if losing but not completely
			modifier = { add = 25 surrender_progress > 0.3 }
			modifier = { add = 15 surrender_progress > 0.5 }
			
			# Democrats prefer negotiation
			modifier = { factor = 2.0 has_government = democratic }
			
			# Exhausted wants any deal
			modifier = { add = 20 has_country_flag = plus_cs_exhausted }
		}
		
		log = "[PEACE_LOG] OUTCOME: [ROOT.GetName] COUNTER-PROPOSED conditional surrender to [FROM.GetName]."
		
		# Send conditional surrender to the victor
		FROM = {
			clr_country_flag = plus_cs_waiting_response
		}
		set_country_flag = { flag = plus_cs_waiting_response days = 1 }
		FROM = { country_event = { id = plus_cs.1 } }
	}
}
